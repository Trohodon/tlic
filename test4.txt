Option Compare Database

Public RawFilePath As String
Public RawFileVersion As Integer
Public Reduced As Boolean
Public AuxFileNumber As Integer
Public AuxFilePath As String

Public Subsystem As Boolean
Public BusSubsys() As Long
Public SubsysSize As Integer

Option Explicit

Public Function BuildSeries() As Integer

    Dim Path As String

    Path = "C:/Cases/"
    ExportSeriesToRaw 31, 2008, "S", Path, True, True, False, True
    BuildSeries = 1

Public Sub ExportSeriesToRaw(RawVersion As Integer, baseYear As Double, BaseSeason As String, Dir As String, AutoDispatch As Boolean, IncludeTielines As Boolean, ReducedNumbering As Boolean, MakeInsertIDEV As Boolean)

    Set db = CurrentDb()

    Reduced = ReducedNumbering
    RawFileVersion = RawVersion

    'Raw info field variables:
    Dim RawType As Integer
    Dim CaseHeader1 As String
    Dim CaseHeader2 As String
    Dim Models As DAO.Recordset
    Dim ModelName As String
    Dim Year As Double
    Dim RetailLoadMW As Double
    Dim ModelIndex As Integer
    Dim Interchange As Double
    Dim Angle As Double

    RawFileVersion = RawVersion

    'TURN OFF WARNINGS!
    DoCmd.SetWarnings False

    CloseAllPSSETables

    Set Models = db.OpenRecordset("Snapshots", dbOpenDynaset, dbSeeChanges)
    Models.MoveLast
    Models.MoveFirst

    While Not Models.EOF

        If Models.Fields("build").Value Then

            ModelName = Models.Fields("model_name").Value
            Year = Models.Fields("year").Value
            RetailLoadMW = Models.Fields("retail_load_mw").Value
            ModelIndex = Models.Fields("model_index").Value
            Interchange = Models.Fields("net_interchange_MW").Value
            Angle = Models.Fields("angle_deg").Value

            If MakeInsertIDEV Then
                RawFilePath = Dir & "sceg" & ModelName & "_insert.idv"
            Else
                RawFilePath = Dir & "sceg" & ModelName & ".raw"
            End If

            RawFileNumber = FreeFile
            Open RawFilePath For Output As #RawFileNumber

    'AUX File:
    AuxFilePath = Dir & "sceg" & ModelName & ".aux"
    AuxFileNumber = FreeFile
    Open AuxFilePath For Output As #AuxFileNumber

    If MakeInsertIDEV Then

        Print #RawFileNumber, "BAT_BSYS,1,0,0.0, 999.,1,343,0,0,0"
        Print #RawFileNumber, "BAT_EXTR,1,0,0,0"
        Print #RawFileNumber, "RDCH"
        Print #RawFileNumber, "1"

    Else

        If IsNull(Models.Fields("header_1").Value) Then
            CaseHeader1 = ""
        Else
            CaseHeader1 = Models.Fields("header_1").Value
        End If

        If IsNull(Models.Fields("header_2").Value) Then
            CaseHeader2 = ""
        Else
            CaseHeader2 = Models.Fields("header_2").Value
        End If

        Print #RawFileNumber, "1,    100.00   /   SCE&G ModelMaster Database " & Now()
        Print #RawFileNumber, CaseHeader1
        Print #RawFileNumber, CaseHeader2

    End If

    If RawVersion = 31 Then

        Print #AuxFileNumber, "DATA Busses(Bus, [BusNum, BusName, BusNomVolt, AreaNum, ZoneNum, BusPUVolt, BusAngle, Latitude, Longitude], AUXCSV)"
			
Print #AuxFileNumber, "{"

ExportBusses ModelIndex, Angle

Print #AuxFileNumber, "}"
Print #AuxFileNumber, ""
Print #AuxFileNumber, "DATA Loads(Load, [BusNum, LoadID, LoadStatus, LoadSMW, LoadSMVR], AUXCSV)"
Print #AuxFileNumber, "{"

ExportLoads baseYear, BaseSeason, Year, RetailLoadMW

Print #AuxFileNumber, "}"
Print #AuxFileNumber, ""
Print #AuxFileNumber, "DATA Generators(Gen, [BusNum, GenID, GenStatus, GenMW, GenMWMax, GenMWMin, GenMVR, GenMVRMax, GenMVRMin, GenVoltSet, GenRegNum, GenAGCAble, GenAVRAble], AUXCSV)"
Print #AuxFileNumber, "{"

ExportFixedShunts ModelIndex

ExportMachines ModelIndex, AutoDispatch, baseYear, BaseSeason, Year, RetailLoadMW

Print #AuxFileNumber, "}"
Print #AuxFileNumber, ""
Print #AuxFileNumber, "DATA Branches(Branch, [BusNum, BusNum:1, LineCircuit, LineStatus, LineR, LineX, LineC, LineAMVA, LineAMVA:1, LineAMVA:2, LineMeter], AUXCSV)"
Print #AuxFileNumber, "{"

ExportBranches ModelIndex, IncludeTielines

Print #AuxFileNumber, "}"
Print #AuxFileNumber, ""
Print #AuxFileNumber, "DATA Transformers(Transformer, [BusNum, BusNum:1, LineCircuit, LineXFType, XFAuto, XFRegBus, XFRegMax, XFRegMin, XFStep, XFTapMax, XFTapMin, LineStatus, LineR, LineX, LineTap, LineAMVA, LineAMVA:1, LineAMVA:2, LineMeter], AUXCSV)"
Print #AuxFileNumber, "{"

ExportTransformers ModelIndex

Print #AuxFileNumber, "}"
Print #AuxFileNumber, ""
Print #AuxFileNumber, "DATA Areas(Area, [AreaNum, AreaName, BusSlack, AreaUnSpecifiedStudyMW, ConvergenceTol], AUXCSV)"
Print #AuxFileNumber, "{"

ExportAreas Interchange
ExportTwoTermDCLines
ExportVoltageSourceConverterDCLines
ExportImpedanceCorrectionTables
ExportMultiTermDCLines
ExportMultisectionLines

Print #AuxFileNumber, "}"

Print #AuxFileNumber, "DATA Zones(Zone, [ZoneNum, ZoneName], AUXCSV)"
Print #AuxFileNumber, "{"

ExportZones

ExportInterareaTransfers
ExportOwners
ExportFACTSDevices

Print #AuxFileNumber, "}"
Print #AuxFileNumber, "DATA Shunts(Shunt, [BusNum, ShuntID, SSStatus, SSCMode, SSVHigh, SSVLow, SSBlockNumSteps, SSBlockMVarPerStep, SSNMVR, SSRegNum], AUXCSV)"
Print #AuxFileNumber, "{"

ExportSwitchedShunts ModelIndex

Print #AuxFileNumber, "}"

End If

If MakeInsertIDEV Then
    Print #RawFileNumber, "QUIT"
End If

Close #RawFileNumber
Close #AuxFileNumber

End If

Models.MoveNext

Wend

'TURN ON WARNINGS!
DoCmd.SetWarnings True

MsgBox "Done"

End Sub

Private Sub ExportBusses(ModelIndex As Integer, Angle As Double)

    'Close bus and fixed shunt tables if open:
    DoCmd.Close acTable, "Busses", acSavePrompt
    DoCmd.Close acTable, "FixedShunts", acSavePrompt

    'Bus field variables:
    Dim i As Long
    Dim bus_name As String
    Dim BASKV As Double
    Dim IDE As Integer
    Dim AREA As Integer
    Dim ZONE As Integer
    Dim GL As Double
    Dim BL As Double
    Dim VM As Double
    Dim VA As Double
    Dim White As Integer

    Dim Longitude As Double
    Dim latitude As Double

    Dim External As Boolean

    Dim FirstModel As Integer
    Dim LastModel As Integer
	
    Dim Active As Boolean
    Dim Include As Boolean
    Dim Tiebus As Boolean
    Dim TiebusNumber As Integer

    Dim Buses As DAO.Recordset
    Set Buses = db.OpenRecordset("BussesWithShunts", dbOpenDynaset, dbSeeChanges)

    'reset cursor:
    Buses.MoveLast
    Buses.MoveFirst

    While Not Buses.EOF

        External = Buses.Fields("external_tie_bus").Value
        Tiebus = Buses.Fields("sceg_tie_bus").Value

        FirstModel = Buses.Fields("first_model_index").Value
        LastModel = Buses.Fields("last_model_index").Value
        Active = ModelIndex >= FirstModel And ModelIndex <= LastModel

        Include = Buses.Fields("include").Value

        If Not External And Active And Include Then

            If Reduced Then
                If Tiebus Then
                    i = Buses.Fields("sceg_tie_bus_number").Value
                Else
                    i = 370000 + Buses.Fields("bus_number").Value
                End If

                bus_name = Trim(Buses.Fields("reduced_name").Value)
            Else
                i = Buses.Fields("bus_number").Value
                bus_name = Trim(Buses.Fields("bus_name").Value)
            End If

            BASKV = Buses.Fields("base_kV").Value

            GL = 0#
            BL = 0#

            If Not IsNull(Buses.Fields("shunt_in_sevice").Value) Then
                If Buses.Fields("shunt_in_sevice").Value Then
                    GL = Buses.Fields("gl_MW").Value
                    BL = Buses.Fields("bl_MW").Value
                End If
            End If

            IDE = Buses.Fields("bus_type_code").Value

            If Reduced Then
                AREA = 343
                ZONE = 1375
            Else
                AREA = Buses.Fields("area_number").Value
				
            ZONE = Buses.Fields("zone_number").Value
        End If

        VM = 1#
        VA = Angle

        Longitude = Buses.Fields("longitude")
        latitude = Buses.Fields("latitude")

        If RawFileVersion = 30 Then

            Print #RawFileNumber, FormatLeftLong(i, 6) & ",'" _
                & FormatLeftStr(bus_name, 12) & "', " _
                & FormatLeft(BASKV, 9, 4) & "," _
                & FormatLeftInt(IDE, 1) & "," _
                & FormatLeft(GL, 10, 3) & "," _
                & FormatLeft(BL, 10, 3) & "," _
                & FormatLeftInt(AREA, 4) & "," _
                & FormatLeftInt(ZONE, 4) & "," _
                & FormatLeft(VM, 7, 5) & "," _
                & FormatLeft(VA, 9, 4)

        End If

        If RawFileVersion = 31 Then

            Print #RawFileNumber, FormatLeftLong(i, 6) & ",'" _
                & FormatLeftStr(bus_name, 12) & "', " _
                & FormatLeft(BASKV, 9, 4) & "," _
                & FormatLeftInt(IDE, 1) & "," _
                & FormatLeftInt(AREA, 4) & "," _
                & FormatLeftInt(ZONE, 4) & ",," _
                & FormatLeft(VM, 7, 5) & "," _
                & FormatLeft(VA, 9, 4)

        End If

        Print #AuxFileNumber, i & ", " & bus_name & ", " & BASKV & ", " & AREA & ", " & ZONE & ", " & VM & ", " & VA & ", " & latitude & ", " & Longitude

    End If

    Buses.MoveNext

Wend

Print #RawFileNumber, "0 / END OF BUS DATA, BEGIN LOAD DATA"

End Sub

Private Sub ExportFixedShunts(ModelIndex As Integer)

    'Close FixedShunts table if open:
    DoCmd.Close acTable, "FixedShunts", acSavePrompt

    'FixedShunts field variables:
	
    Dim i As Long
    Dim ID As String
    Dim Status As Integer
    Dim GL As Double
    Dim BL As Double

    Dim FixedShunts As DAO.Recordset

    Set FixedShunts = db.OpenRecordset("FixedShunts", dbOpenDynaset, dbSeeChanges)

    'reset cursor:
    FixedShunts.MoveLast
    FixedShunts.MoveFirst

    If RawFileVersion = 30 Then

        'do nothing - fixed shunt data attached to bus records in version 30

    ElseIf RawFileVersion = 31 Then

        While Not FixedShunts.EOF

            i = FixedShunts.Fields("bus_number").Value
            ID = FixedShunts.Fields("shunt_id").Value

            Status = 1
            If FixedShunts.Fields("in_service").Value = False Then
                Status = 0
            End If

            GL = FixedShunts.Fields("gl_MW").Value
            BL = FixedShunts.Fields("bl_MW").Value

            Print #RawFileNumber, FormatLeftLong(i, 6) & ",'" _
                & FormatLeftStr(ID, 2) & "', " _
                & FormatLeftInt(Status, 2) & "," _
                & FormatLeft(GL, 10, 2) & "," _
                & FormatLeft(BL, 10, 2)

            FixedShunts.MoveNext

        Wend

        Print #RawFileNumber, "0 / END OF BUS SHUNT DATA, BEGIN LOAD DATA"

    End If

End Sub

Private Sub ExportLoads(baseYear As Double, BaseSeason As String, CurrentYear As Double, RetailMW As Double)

    'Close Load table if open:
    DoCmd.Close acTable, "Loads", acSavePrompt

    Dim QDF As DAO.QueryDef
    Dim BaseLoadsQ As DAO.QueryDef
	
    Dim LoadForecastWindowedQ As DAO.QueryDef
    Dim CummulativeLoadIncrementsQ As DAO.QueryDef
    Dim Loads As DAO.Recordset

    'Load field variables:
    Dim i As Long
    Dim ID As String
    Dim Status As Integer
    Dim STATUS_STR As String
    Dim AREA As Integer
    Dim ZONE As Integer
    Dim PL As Double
    Dim QL As Double

    Set QDF = db.QueryDefs("ScaledCurrentLoads")
    QDF.Parameters("BaseYear") = baseYear
    QDF.Parameters("BaseSeason") = BaseSeason
    QDF.Parameters("CurrentYear") = CurrentYear
    QDF.Parameters("RetailMW") = RetailMW

    Set Loads = QDF.OpenRecordset

    If Loads.BOF And Loads.EOF Then
        'do nothing, no loads
    Else

        'reset cursor:
        Loads.MoveLast
        Loads.MoveFirst

        While Not Loads.EOF

            If Reduced Then
                If Loads.Fields("sceg_tie_bus").Value Then
                    i = Loads.Fields("sceg_tie_bus_number").Value
                Else
                    i = 370000 + Loads.Fields("pti_bus").Value
                End If
            Else
                i = Loads.Fields("pti_bus").Value
            End If

            ID = Loads.Fields("pti_load").Value
            Status = 1
            STATUS_STR = "Yes"
            If Loads.Fields("in_service").Value = 0 Then
                Status = 0
                STATUS_STR = "No"
            End If

            'If Reduced Then
            '    area = 343
            '    zone = 1375
            'Else
            '    area = Loads.Fields("area_number").Value
            '    zone = Loads.Fields("zone_number").Value	
            'End If

            PL = Loads.Fields("P").Value
            QL = Loads.Fields("Q").Value

            If QL < 0.1 Then
                QL = 0#
            End If

            Print #RawFileNumber, FormatLeftLong(i, 6) & ",'" _
                & FormatLeftStr(ID, 2) & "'," _
                & FormatLeftInt(Status, 1) & "," _
                & "," _
				& "," _
                & FormatLeft(PL, 10, 2) & "," _
                & FormatLeft(QL, 10, 2)

            Print #AuxFileNumber, i & ", " & ID & ", " & STATUS_STR & ", " & PL & ", " & QL

            Loads.MoveNext

        Wend

    End If

    'SRS hard coded for now, need to make an override load field or something:
    Print #RawFileNumber, "  972,'1 ',1,,,     2.16,     2.20"
    Print #RawFileNumber, "  922,'1 ',1,,,     6.98,     3.65"
    Print #RawFileNumber, "  913,'1 ',1,,,     0.71,     0.31"
    Print #RawFileNumber, "  921,'1 ',1,,,     3.06,     2.20"
    Print #RawFileNumber, "  915,'1 ',1,,,     1.10,     1.76"
    Print #RawFileNumber, "  914,'1 ',1,,,     0.35,     0.00"
    Print #RawFileNumber, "  951,'1 ',1,,,     0.82,     0.67"
    Print #RawFileNumber, "  941,'1 ',1,,,     1.22,     0.94"
    Print #RawFileNumber, "  971,'1 ',1,,,     3.61,     3.33"
    Print #RawFileNumber, "0 / END OF LOAD DATA, BEGIN GENERATOR DATA"

End Sub

Private Sub ExportMachines(ModelIndex As Integer, AutoDispatch As Boolean, baseYear As Double, BaseSeason As String, CurrentYear As Double, RetailMW As Double)

    'Close machines table if open:
    DoCmd.Close acQuery, "MachinesWithPlants", acSavePrompt

    'Machine field variables:
    Dim i As Long
    Dim ID As String
    Dim PG As Double
    Dim QG As Double
    Dim QT As Double
    Dim QB As Double
    Dim VS As Double
    Dim IREG As Long
    Dim Mbase As Double
    Dim ZR As Double
    Dim ZX As Double
    Dim RT As Double

    Dim XT As Double
    Dim GTAP As Double
    Dim Stat As Integer
    Dim STATUS_STR As String
    Dim RMPCT As Double
    Dim PT As Double
    Dim PB As Double
    Dim O1 As Integer
    Dim F1 As Double
    Dim Record As String

    Dim QDF As DAO.QueryDef
    Dim Machines As DAO.Recordset
    Dim LoadLossGenTotals As DAO.Recordset
    Dim TotalGenMW As Double
    Dim GenMW As Double
    Dim IsSwing As Boolean
    Dim SwingMaxMW As Double
    Dim GenMet As Boolean

    SwingMaxMW = 70#

    Set QDF = db.QueryDefs("LoadLossGenTotals")
    QDF.Parameters("BaseYear") = baseYear
    QDF.Parameters("BaseSeason") = BaseSeason
    QDF.Parameters("CurrentYear") = CurrentYear
    QDF.Parameters("RetailMW") = RetailMW

    Set LoadLossGenTotals = QDF.OpenRecordset
    LoadLossGenTotals.MoveFirst
    TotalGenMW = LoadLossGenTotals.Fields("total_gen_MW").Value
    GenMW = 0#

    'Set QDF = Db.QueryDefs("CurrentAvailableGen")
    'QDF.Parameters("model_index") = ModelIndex
    'Set Machines = QDF.OpenRecordset

    Set Machines = db.OpenRecordset("Machines", dbOpenDynaset, dbSeeChanges)

    'reset cursor:
    Machines.MoveLast
    Machines.MoveFirst

    While Not Machines.EOF

        i = Machines.Fields("bus_number").Value

        If Reduced Then
            If Machines.Fields("sceg_tie_bus1").Value Then
                i = Machines.Fields("sceg_tie_bus_number1").Value
            Else
                i = 370000 + Machines.Fields("bus_number").Value
            End If
        Else
		
            i = Machines.Fields("bus_number").Value
        End If

        ID = Machines.Fields("machine_id").Value
        PG = 0#
        QG = 0#
        QT = Machines.Fields("qmax_MVAR").Value
        QB = Machines.Fields("qmin_MVAR").Value
        VS = Machines.Fields("v_sched_pu").Value

        If Machines.Fields("remote_bus").Value = 0 Then
            IREG = 0
        Else
            IREG = Machines.Fields("remote_bus").Value
        End If

        If Reduced Then
            If Machines.Fields("sceg_tie_bus2").Value Then
                IREG = Machines.Fields("sceg_tie_bus_number2").Value
            Else
                If Machines.Fields("remote_bus").Value = 0 Then
                    IREG = 0
                Else
                    IREG = 370000 + Machines.Fields("remote_bus").Value
                End If
            End If
        Else
            If Machines.Fields("remote_bus").Value = 0 Then
                IREG = 0
            Else
                IREG = Machines.Fields("remote_bus").Value
            End If
        End If

        Mbase = Machines.Fields("mbase_MVA").Value
        ZR = Machines.Fields("r_source_pu").Value
        ZX = Machines.Fields("x_source_pu").Value
        RT = 0#
        XT = 0#
        GTAP = 1#

        Stat = 1
        STATUS_STR = "Yes"
        If Machines.Fields("available").Value = 0 Then
            Stat = 0
            STATUS_STR = "No"
        End If

        RMPCT = Machines.Fields("rmpct_percent").Value
        PT = Machines.Fields("pmax_MW").Value
        PB = Machines.Fields("pmin_MW").Value
        O1 = 1#
        F1 = 1#

        IsSwing = Machines.Fields("is_slack").Value
		
    ' Autodispatch Routine:
    If AutoDispatch Then
        If IsSwing Then
            PG = TotalGenMW - GenMW
        Else
            If GenMet Then
                Stat = 0
                PG = 0#
            Else
                GenMW = GenMW + PG
                If GenMW + SwingMaxMW > TotalGenMW Then
                    GenMet = True
                End If
            End If
        End If
    End If

    Record = ""

    Record = Record & FormatLeftLong(i, 6) & ","
    Record = Record & "'" & FormatLeftStr(ID, 2) & "'" & ","
    Record = Record & FormatLeft(PG, 10, 3) & ","
    Record = Record & FormatLeft(QG, 10, 3) & ","
    Record = Record & FormatLeft(QT, 10, 3) & ","
    Record = Record & FormatLeft(QB, 10, 3) & ","
    Record = Record & FormatLeft(VS, 7, 5) & ","
    Record = Record & FormatLeftLong(IREG, 6) & ","
    Record = Record & FormatLeft(Mbase, 10, 3) & ","
    Record = Record & FormatLeft(ZR, 10, 5) & ","
    Record = Record & FormatLeft(ZX, 10, 5) & ","
    Record = Record & FormatLeft(RT, 10, 5) & ","
    Record = Record & FormatLeft(XT, 10, 5) & ","
    Record = Record & FormatLeft(GTAP, 7, 5) & ","
    Record = Record & FormatLeftInt(Stat, 2) & ","
    Record = Record & FormatLeft(RMPCT, 8, 1) & ","
    Record = Record & FormatLeft(PT, 10, 3) & ","
    Record = Record & FormatLeft(PB, 10, 3) & ","
    Record = Record & FormatLeftInt(O1, 4) & ","
    Record = Record & FormatLeft(F1, 6, 4)

    Print #RawFileNumber, Record

    Print #AuxFileNumber, i & ", " & ID & ", " & STATUS_STR & ", " & PG & ", " & PT & ", " & PB & ", " & QG & ", " & QT & ", " & QB & ", " & VS & ", " & IREG & ", Yes, Yes"

    Machines.MoveNext

Wend

'MsgBox "Swing: " & (TotalGenMW - GenMW)

Print #RawFileNumber, "0 / END OF GENERATOR DATA, BEGIN BRANCH DATA"

End Sub

Private Sub ExportBranches(ModelIndex As Integer, IncludeTielines As Boolean)

    'Close branch table if open:
    DoCmd.Close acTable, "Branches", acSavePrompt

    'Branch field variables:
    Dim i As Long
    Dim j As Long
    Dim Ckt As String
    Dim R As Double
    Dim x As Double
    Dim b As Double
    Dim RateA As Double
    Dim RateB As Double
    Dim RateC As Double
    Dim GI As Double
    Dim bi As Double
    Dim GJ As Double
    Dim BJ As Double
    Dim ST As Integer
    Dim STATUS_STR As String
    Dim MET As Integer
    Dim METERED_STR As String
    Dim Length As Double
    Dim O1 As Integer
    Dim F1 As Double
    Dim Record As String
    Dim External As Boolean

    Dim Include As Boolean
    Dim IgnoreDates As Boolean

    Dim FirstModel As Integer
    Dim LastModel As Integer
    Dim Active As Boolean

    Dim Branches As DAO.Recordset
    Dim Tielines As DAO.Recordset

    Set Branches = db.OpenRecordset("NonTransformerBranches", dbOpenDynaset, dbSeeChanges)

    'reset cursor:
    Branches.MoveLast
    Branches.MoveFirst

    While Not Branches.EOF

        FirstModel = Branches.Fields("first_model_index").Value
        LastModel = Branches.Fields("last_model_index").Value

        Active = False
        If Branches.Fields("include").Value Then
            If Not Branches.Fields("ignore_dates").Value Then
                Active = ModelIndex >= FirstModel And ModelIndex <= LastModel
            End If
        End If
		
        If Active Then

            If Reduced Then
                If Branches.Fields("sceg_tie_bus1").Value Then
                    i = Branches.Fields("sceg_tie_bus_number1").Value
                ElseIf Branches.Fields("external1").Value Then
                    i = Branches.Fields("bus1").Value
                Else
                    i = 370000 + Branches.Fields("bus1").Value
                End If
            Else
                i = Branches.Fields("bus1").Value
            End If

            If Reduced Then
                If Branches.Fields("sceg_tie_bus2").Value Then
                    j = Branches.Fields("sceg_tie_bus_number2").Value
                ElseIf Branches.Fields("external2").Value Then
                    j = Branches.Fields("bus2").Value
                Else
                    j = 370000 + Branches.Fields("bus2").Value
                End If
            Else
                j = Branches.Fields("bus2").Value
            End If

            If IsNull(Branches.Fields("ckt").Value) Then
                Ckt = "1"
            Else
                Ckt = Branches.Fields("ckt").Value
            End If

            If IsNull(Branches.Fields("Rpu").Value) Then
                R = 0.001
            Else
                R = Branches.Fields("Rpu").Value
            End If

            If IsNull(Branches.Fields("Xpu").Value) Then
                x = 0.01
            Else
                x = Branches.Fields("Xpu").Value
            End If

            If IsNull(Branches.Fields("Bpu").Value) Then
                b = 0.002
            Else
                b = Branches.Fields("Bpu").Value
            End If

            If IsNull(Branches.Fields("RateA").Value) Then
                RateA = 999
            Else
                RateA = Branches.Fields("RateA").Value
            End If

            If IsNull(Branches.Fields("RateB").Value) Then
                RateB = 999
            Else
                RateB = Branches.Fields("RateB").Value
            End If

            If IsNull(Branches.Fields("RateC").Value) Then
                RateC = 999
            Else
                RateC = Branches.Fields("RateC").Value
            End If

            ST = 1
            STATUS_STR = "Yes"
            If Branches.Fields("is_normally_closed").Value = 0 Then
                ST = 0
                STATUS_STR = "No"
            End If

            If IsNull(Branches.Fields("metered_end").Value) Then
                MET = 1
            Else
                MET = Branches.Fields("metered_end").Value
            End If

            METERED_STR = "From"
            If MET = 2 Then
                METERED_STR = "To"
            End If

            If IsNull(Branches.Fields("mileage").Value) Then
                Length = 0#
            Else
                Length = Branches.Fields("mileage").Value
            End If

            O1 = 1
            F1 = 1#

            Record = ""
            Record = Record & FormatLeftLong(i, 6) & ","
            Record = Record & FormatLeftLong(j, 7) & ","
            Record = Record & "'" & FormatLeftStr(Ckt, 2) & "'" & ","
            Record = Record & FormatLeft(R, 10, 5) & ","
            Record = Record & FormatLeft(x, 10, 5) & ","
            Record = Record & FormatLeft(b, 10, 5) & ","
            Record = Record & FormatLeft(RateA, 8, 2) & ","
            Record = Record & FormatLeft(RateB, 8, 2) & ","
            Record = Record & FormatLeft(RateC, 8, 2) & ",,,,,"
            Record = Record & FormatLeftInt(ST, 1) & ","
			
            If RawFileVersion = 31 Then
                Record = Record & FormatLeftInt(MET, 1) & ","
            End If

            Record = Record & FormatLeft(Length, 7, 2) & ","
            Record = Record & FormatLeftInt(O1, 4) & ","
            Record = Record & FormatLeft(F1, 6, 4)

            Print #RawFileNumber, Record

            Print #AuxFileNumber, i & ", " & j & ", " & Ckt & ", " & STATUS_STR & ", " & R & ", " & x & ", " & b & ", " & Round(RateA, 1) & ", " & Round(RateB, 1) & ", " & Round(RateC, 1) & ", " & METERED_STR

        End If

        Branches.MoveNext

    Wend

    Print #RawFileNumber, "0 / END OF BRANCH DATA, BEGIN TRANSFORMER DATA"

Public Function FormatLeft(num As Double, Width As Integer, Precision As Integer)

    Dim DecFormat As String
    Dim i As Integer

    DecFormat = "0."

    While i < Precision
        DecFormat = DecFormat & "0"
        i = i + 1
    Wend

    FormatLeft = Space(Width - Len(Format(num, DecFormat))) & Format(num, DecFormat)

End Function

Public Function FormatLeftInt(num As Integer, Width As Integer)
    FormatLeftInt = Space(Width - Len(Format(num, "0"))) & num
End Function

Public Function FormatLeftLong(num As Long, Width As Integer)
    FormatLeftLong = Space(Width - Len(Format(num, "0"))) & num
End Function

Public Function FormatLeftStr(str As String, Width As Integer)
    If (Width - Len(str)) >= 0 Then
        FormatLeftStr = str & Space(Width - Len(str))
    Else
        FormatLeftStr = str
    End If
End Function

Public Sub WritePWGMDData(FilePath As String, SnapshotIndex As Integer, busOffset As Long, combineBuses As Boolean, percentOfSolar As Integer, ignoreAfterQueueNum As Integer, _
incNoIA As Boolean, Dynamics As Boolean, inc46kvData As Boolean, incSolar As Boolean, useRegionalNumbering As Boolean)

    Dim QDF As DAO.QueryDef
    Dim SQL As String

    Dim Buses As DAO.Recordset
    Dim Substation As DAO.Recordset
    Dim Transformers As DAO.Recordset

    Set db = CurrentDb()

    Dim Snapshots As DAO.Recordset
    Dim QueuedProjects As DAO.Recordset
    Dim SnapshotName As String
    Dim SnapshotYear As Long
    Dim SnapShotSeason As String
    Dim snapshotDate As Date
    Dim Line As String
    Dim Bus1 As Long 'for xfmr data
    Dim Bus2 As Long 'for xfmr data
    Dim incQueued As Boolean

    'get snapshot info:
    Set Snapshots = db.OpenRecordset("Snapshots", dbOpenDynaset, dbSeeChanges)
    Snapshots.MoveLast
    Snapshots.MoveFirst
    While Not Snapshots.EOF
        If Snapshots.Fields("model_index") = SnapshotIndex Then
            SnapshotName = Snapshots.Fields("model_name")
            SnapshotYear = Snapshots.Fields("year")
            SnapShotSeason = Snapshots.Fields("season")
            snapshotDate = Snapshots.Fields("date")
        End If
        Snapshots.MoveNext
    Wend
    Snapshots.Close

    Dim maxBus As Long
    maxBus = 3000
    If inc46kvData Then _
        maxBus = 5000
		
    'AUX File:
    AuxFilePath = FilePath & "_GMD" & ".aux"
    AuxFileNumber = FreeFile
    Open AuxFilePath For Output As #AuxFileNumber

    'Begin writing .aux file
    Print #AuxFileNumber, "//This file contains substation info for GMD Studies"
    Print #AuxFileNumber, "// Auto-Generated Snapshot from SCEGModelData DB"
    Print #AuxFileNumber, "// File Created: " & Now()
    Print #AuxFileNumber, "// Snapshot Year: " & SnapshotYear
    Print #AuxFileNumber, "// Snapshot Season: " & SnapShotSeason
    Print #AuxFileNumber, "// Snapshot Date: " & snapshotDate

    'Start Substation data
    Print #AuxFileNumber, "//===================== SUBSTATION DATA ========================"
    Print #AuxFileNumber, ""
	Print #AuxFileNumber, ""
    Print #AuxFileNumber, "//Start Substation data.  The substation data does not honor a snapshot date, it just outputs all substations in the substations table."
    Print #AuxFileNumber, "DATA (SUBSTATION, [SubNum,Longitude,Latitude,GICSubGroundOhms, SubName],,YES)"
    Print #AuxFileNumber, "{"

    Set Substation = db.OpenRecordset("Substation", dbOpenDynaset, dbSeeChanges)
    Substation.MoveLast
    Substation.MoveFirst

    While Not Substation.EOF
        If Substation.Fields("Include") = "True" Then ' Or Substation.Fields("status") = "Future" Then
            Line = Format(Substation.Fields("sub id"), "000000")
            Line = Line & "  " & Format(Substation.Fields("longitude"), "0.00000000")
            Line = Line & "  " & Format(Substation.Fields("latitude"), "0.00000000")
            Line = Line & "  " & Format(Substation.Fields("Grounding Resistance (Ohms)"), "0.0000")
            Line = Line & "  " & Chr(34) & Substation.Fields("sub name") & Chr(34)  ' Chr(34) is a quotation

            Print #AuxFileNumber, Line
        Else
        'Substation.MoveNext
        End If
        Substation.MoveNext
    Wend
    Substation.Close
    Print #AuxFileNumber, "}"

    'Start bus data
    Print #AuxFileNumber, "//===================== BUS DATA ========================"
    Print #AuxFileNumber, "//Start Bus data.  This data honors the snapshot date, it assigns all buses to a substation."
    Print #AuxFileNumber, "DATA (BUS, [BusNum,SubNum])"
    Print #AuxFileNumber, "{"

    Set Buses = db.OpenRecordset("Buses", dbOpenDynaset, dbSeeChanges)
    Buses.MoveLast
    Buses.MoveFirst
    While Not Buses.EOF
        If Buses.Fields("Include") And Not Buses.Fields("bus_number") = 0 And BusInSubsystem(Buses.Fields("bus_number")) Then
            Set QueuedProjects = db.OpenRecordset("QueuedProjects", dbOpenDynaset, dbSeeChanges)
            incQueued = False
			
            If incSolar And Not (IsNull(Buses.Fields("Queue"))) Then
                If Buses.Fields("Queue") <= ignoreAfterQueueNum And incNoIA Then
                    incQueued = True
                ElseIf Buses.Fields("Queue") <= ignoreAfterQueueNum Then
                    QueuedProjects.MoveLast
                    QueuedProjects.MoveFirst

                    While Not QueuedProjects.EOF And Buses.Fields("Queue") <> QueuedProjects.Fields("QueueNum")
                        QueuedProjects.MoveNext
                    Wend

                    If Not QueuedProjects.EOF And QueuedProjects.Fields("SignedIA") Then
                        incQueued = True
                    End If

                    QueuedProjects.Close
                End If
            End If

            If (Not combineBuses Or Buses.Fields("parent_bus_number") = 0) And (IsNull(Buses.Fields("Queue")) Or (Not IsNull(Buses.Fields("Queue")) And incQueued)) Then
                If Buses.Fields("start_date") <= snapshotDate And Buses.Fields("end_date") > snapshotDate Then
                    If Buses.Fields("bus_number") < maxBus And (Dynamics And (Buses.Fields("bus_number") < 740 And Buses.Fields("bus_number") > 722) _
					    Or (Buses.Fields("bus_number") >= 740 Or Buses.Fields("bus_number") <= 722)) Then ' VCS1 SS buses between 723 and 740
                        'If Buses.Fields("external_tie_bus") Then
                        '    Line = "psspy.bus_data_3(" & Buses.Fields("bus_number") & ", [" 'do not put external busses into python file
                        'Else
                        Line = Buses.Fields("bus_number") + busOffset
                        Line = Line & "  " & Format(Buses.Fields("sub_id"), "0000")
                        Line = Line & "    //" & Buses.Fields("long_name")

                        Print #AuxFileNumber, Line
                        'End If
                    End If
                End If
            End If

        End If

        Buses.MoveNext

    Wend

    Buses.Close
    Print #AuxFileNumber, "}"

    'Start transformer data
    Print #AuxFileNumber, "//===================== TRANSFORMER DATA ========================"
    Print #AuxFileNumber, "//Start Transformer data.  This data honors the snapshot date, it assigns GMD data to transformers."
    Print #AuxFileNumber, "GICXFormer (BusNum3W,BusNum3W:1,BusNum3W:2,LineCircuit,GICManualCoilR,GICXFCoilR1,"
    Print #AuxFileNumber, "                 GICXFCoilR1:1,XFConfiguration,XFConfiguration:1,XFIsAutoXF,XFCoreType)"
    Print #AuxFileNumber, "{"

    Set Transformers = db.OpenRecordset("TwoWindingsWithBusses", dbOpenDynaset, dbSeeChanges)
    Transformers.MoveLast
    Transformers.MoveFirst

    While Not Transformers.EOF
        If Not IsNull(Transformers.Fields("from_bus")) And Not IsNull(Transformers.Fields("to_bus")) Then
            Bus1 = Transformers.Fields("from_bus")
            Bus2 = Transformers.Fields("to_bus")
			
            If Transformers.Fields("GMD_Applicable") And Transformers.Fields("start_date") <= snapshotDate And Transformers.Fields("end_date") > snapshotDate Then
                incQueued = False

                If incSolar And Not (IsNull(Transformers.Fields("Queue"))) Then
                    If Transformers.Fields("Queue") <= ignoreAfterQueueNum And incNoIA Then
                        incQueued = True
                    ElseIf Transformers.Fields("Queue") <= ignoreAfterQueueNum Then
                        QueuedProjects.MoveLast
                        QueuedProjects.MoveFirst

                        While Not QueuedProjects.EOF And Transformers.Fields("Queue") <> QueuedProjects.Fields("QueueNum")
                            QueuedProjects.MoveNext
                        Wend

                        If Not QueuedProjects.EOF And QueuedProjects.Fields("SignedIA") Then
                            incQueued = True
                        End If

                        QueuedProjects.Close
                    End If
                End If

                If BusInSubsystem(Bus1) And BusInSubsystem(Bus2) And (IsNull(Transformers.Fields("Queue")) Or (Not IsNull(Transformers.Fields("Queue")) And incQueued)) Then
                    If Not Transformers.Fields("external_1") And useRegionalNumbering Then _
                        Bus1 = Bus1 + busOffset
                    If Not Transformers.Fields("external_2") And useRegionalNumbering Then _
                        Bus2 = Bus2 + busOffset

                    Line = Bus1 & " " & Bus2 & " "
                    Line = Line & Transformers.Fields("GMD_3WindBusNum") & " "
                    Line = Line & Chr(34) & Trim(Transformers.Fields("ckt")) & Chr(34) & " "

                    If Transformers.Fields("GMD_CoilResAvail") Then
                        Line = Line & Chr(34) & "Yes" & Chr(34) & " "
                    Else
                        Line = Line & Chr(34) & "No" & Chr(34) & " "

                    Line = Line & Format(Transformers.Fields("GMD_XFMR_HS_Res"), "0.000000") & "  "
                    Line = Line & Format(Transformers.Fields("GMD_XFMR_LS_Res"), "0.000000") & "  "
                    Line = Line & Chr(34) & Transformers.Fields("GMD_XFMR_HS_Config") & Chr(34) & "  "
                    Line = Line & Chr(34) & Transformers.Fields("GMD_XFMR_LS_Config") & Chr(34) & "  "

                    If Transformers.Fields("is_autotransformer") Then
                        Line = Line & Chr(34) & "Yes" & Chr(34) & " "
                    Else
                        Line = Line & Chr(34) & "No" & Chr(34) & " "
                    End If

                    Line = Line & Chr(34) & Transformers.Fields("GMD_XFMR_CoreType") & Chr(34)

                    'add comments to file
                    Dim fromBus As String, toBus As String
                    fromBus = Transformers.Fields("bus_name_1")

            Line = Line & "    // " & fromBus & "-->" & toBus & " ckt " & Trim(Transformers.Fields("ckt"))
            Print #AuxFileNumber, Line

        End If
    End If 'If Transformers.Fields("GMD_Applicable") And Transformers.Fields("start_date") <= SnapshotDate And Transformers.Fields("end_date") > SnapshotDate Then
    End If 'If Not IsNull(Transformers.Fields("from_bus")) And Not IsNull(Transformers.Fields("to_bus")) Then

Print #AuxFileNumber, "}"

	    Print #File, "// Snapshot Date: " & snapshotDate

                        '(IsNull(filteredBranch.Fields("Queue")) Or (incSolar And filteredBranch.Fields("IfQueueNotPres") = "" And filteredBranch.Fields("Queue") <= ignoreAfterQueueNum And _

                            busTo = filteredBranch.Fields("bus2")

                            filteredBranch.Fields("bus1Name") & " TO " & filteredBranch.Fields("bus2Name")

                    filteredBranch.MoveNext

                If foundData Then _
                    Print #File, branchString

                filteredBranch.Close

            End If

            If foundData Then _
                Print #File, vbTab & "</SUBDATA>"

        End If

        Line.MoveNext

    Wend

    Line.Close

    On Error Resume Next
    db.QueryDefs.Delete "filteredBranch"
    On Error GoTo 0

    Print #File, "}"
	
    Close #File

    'create device data aux file:
    File = FreeFile
    If Dir$(FilePath & "_DeviceData.aux") <> "" Then _
        Kill (FilePath & "_DeviceData.aux")
    Open FilePath & "_DeviceData.aux" For Output As #File

    'print file metadata:
    Print #File, "// This file contains data for placing the device numbers on the oneline powerworld display.  Each of the 4 filters call this file at the bottom."
    Print #File, "// It lists SCEG's lines to help distingish outages"

    Print #File, "// Snapshot Year: " & SnapshotYear

            Else
                busTo = DeviceData.Fields("bus2")

            branchString = branchString & """" & DeviceData.Fields("nostat") & """ """

            Dim frFirst As String, toFirst As String
            frFirst = Left(DeviceData.Fields("fromnum"), 1)
            toFirst = Left(DeviceData.Fields("tonum"), 1)

            If frFirst = "B" Or frFirst = "C" Or frFirst = "H" Or frFirst = "L" Or frFirst = "S" Then
                branchString = branchString & Right(DeviceData.Fields("fromnum"), Len(DeviceData.Fields("fromnum")) - 2)

                branchString = branchString & DeviceData.Fields("fromnum")
            End If

            If toFirst = "B" Or toFirst = "C" Or toFirst = "H" Or toFirst = "L" Or toFirst = "S" Then
			
                branchString = branchString & """ """ & Right(DeviceData.Fields("tonum"), Len(DeviceData.Fields("tonum")) - 2)
            Else
                branchString = branchString & """ """ & DeviceData.Fields("tonum")
            End If

            branchString = branchString & """ """ & DeviceData.Fields("fromtype") & """ """ & _
            DeviceData.Fields("totype") & """ " & busFr & " " & busTo & " """ & DeviceData.Fields("ckt") & """" & vbTab & vbTab & vbTab & vbTab & vbTab & "// """ & DeviceData.Fields("fromnum") & """ """ & _
            DeviceData.Fields("tonum") & """ """ & DeviceData.Fields("bus1Name") & """ """ & DeviceData.Fields("bus2Name") & """ """ & DeviceData.Fields("line_name") & """"

            Print #File, branchString

            branchString = ""

        End If

        DeviceData.MoveNext

    Wend

    DeviceData.Close

    'Transformer device data
    Set DeviceData = db.OpenRecordset("TFMR_DeviceData", dbOpenDynaset, dbSeeChanges)
    DeviceData.MoveLast
    DeviceData.MoveFirst

    branchString = ""
    While Not DeviceData.EOF
        If DeviceData.Fields("start_date") <= snapshotDate And snapshotDate <= DeviceData.Fields("end_date") Then
            If DeviceData.Fields("from_bus") < 9999 Then
                busFr = DeviceData.Fields("from_bus") + busOffset
            Else
                busFr = DeviceData.Fields("to_bus")
            End If

            If DeviceData.Fields("to_bus") < 9999 Then
                busTo = DeviceData.Fields("to_bus") + busOffset
            Else
                busTo = DeviceData.Fields("to_bus")

            ' "nostat" "fromnum" "tonum" "fromtype" "totype" frombus tobus "cktid" tabX5 // "frombusname" "tobusname"
            branchString = branchString & """" & DeviceData.Fields("nostat") & """ """

            frFirst = Left(DeviceData.Fields("fromnum"), 1)
            toFirst = Left(DeviceData.Fields("tonum"), 1)

                branchString = branchString & DeviceData.Fields("fromnum")
            End If

            If toFirst = "B" Or toFirst = "C" Or toFirst = "H" Or toFirst = "L" Or toFirst = "S" Then
				branchString = branchString & """ """ & Right(DeviceData.Fields("tonum"), Len(DeviceData.Fields("tonum")) - 2)
			Else
				branchString = branchString & """ """ & DeviceData.Fields("tonum")
			End If

			branchString = branchString & """ """ & DeviceData.Fields("fromtype") & """ """ & _
            DeviceData.Fields("totype") & """ " & busFr & " " & busTo & " """ & DeviceData.Fields("ckt") & """" & vbTab & vbTab & vbTab & vbTab & vbTab & "// """ & DeviceData.Fields("from_bus_name") & """ """ & _
            DeviceData.Fields("to_bus_name") & """ """ & DeviceData.Fields("line_name") & """"
			Print #File, branchString
			
			branchString = ""
		End If
		
		DeviceData.MoveNext
	Wend

	DeviceData.Close

	Print #File, NOneighbor   ' Print out N.O. data for neighbors
	Print #File, "}"
	Print #File, "//--------------------------------------------------------------------------------"
    Print #File, "// All caps with device numbers"
    Print #File, "//now called up by shuntdata.aux"

    Close #File

    'create Common Structure (BREAKER TO BREAKER) contingency file based on new Table in DB
    'CSL1 query SQL: SELECT CommonStructureCkts.ID, CommonStructureCkts.Line1Name, Line.line_id, Line.start_date, Line.active_start, Line.end_date, Line.InclAsCTG, Line.Queue, Line.NotQueue FROM Line INNER JOIN CommonStructureCkts ON Line.line_name = CommonStructureCkts.Line1Name;
    'CSL2 query SQL: SELECT CommonStructureCkts.ID, CommonStructureCkts.Line2Name, Line.start_date, Line.active_start, Line.end_date, Line.InclAsCTG, Line.Queue, Line.NotQueue, Line.line_id FROM Line INNER JOIN CommonStructureCkts ON Line.line_name = CommonStructureCkts.Line2Name;
    File_CSB2B = FreeFile
    If Dir$(FilePath & "_NONcomb_CSB2B.aux") <> "" Then _
        Kill (FilePath & "_NONcomb_CSB2B.aux")
    Open FilePath & "_NONcomb_CSB2B.aux" For Output As #File_CSB2B

    Print #File_CSB2B, "// PowerWorld NONCOMBINABLE Common Structure BREAKER-2-BREAKER Contingency Data for SCEG Control Area"
    Print #File_CSB2B, "// Auto-Generated Snapshot from SCEGModelData DB"
    Print #File_CSB2B, "// File Created: " & Now()
    Print #File_CSB2B, "// Snapshot Year: " & SnapshotYear
    Print #File_CSB2B, "// Snapshot Season: " & SnapShotSeason
    Print #File_CSB2B, "// Snapshot Date: " & snapshotDate

    Print #File_CSB2B, "{"

    'Open new tables and queries
    Dim CSCTG As DAO.Recordset
    'Dim CSL1Query As DAO.QueryDef
    'Dim CSL2Query As DAO.QueryDef

    Dim CSL2 As DAO.Recordset
    Dim branchFilter1 As DAO.Recordset

    'Set CSL1 = db.OpenRecordset("CSL1", dbOpenDynaset, dbSeeChanges)
    'Set CSL2 = db.OpenRecordset("CSL2", dbOpenDynaset, dbSeeChanges)
    CSCTG.MoveLast
    CSCTG.MoveFirst

    'determine line_id numbers for each Common Structure contingency defined in the table
    Dim CSCTG_id As Integer, line1_id As Integer, line2_id As Integer
	
    Dim line1Found As Boolean, line2Found As Boolean
    While Not CSCTG.EOF 'reads through entire table

        'check to make sure the contingency entry on the table is included and within valid dates
        If CSCTG.Fields("Include") And CSCTG.Fields("StartDate") <= snapshotDate And snapshotDate < CSCTG.Fields("EndDate") Then
            CSCTG_id = CSCTG.Fields("ID")
            Print #File_CSB2B, vbNewLine & "// CS CTG ID#: " & CSCTG_id
            line1Found = False
            line2Found = False

            'Line 1
            'create Common Structure Line 1 query
            On Error Resume Next
            db.QueryDefs.Delete "CSL1"
            On Error GoTo 0
            SQL = "SELECT * FROM Line INNER JOIN CommonStructureCkts ON Line.line_name = CommonStructureCkts.Line1Name WHERE (((CommonStructureCkts.ID)=" & CSCTG_id & "))"
            Set CSL1 = db.OpenRecordset(SQL, dbOpenSnapshot)
            CSL1.MoveLast
            CSL1.MoveFirst

            'for each line, evaluate if the date range is valid, it's included as a CTG, and if the queue dependency logic blocks the line or not
            While Not CSL1.EOF And Not line1Found
                If CSL1.Fields("start_date") <= snapshotDate And snapshotDate < CSL1.Fields("end_date") And CSL1.Fields("InclAsCTG") And assessQueueDependency(CSL1.Fields("line_id"), incNoIA, ignoreAfterQueueNum) Then
                    'valid line found for that CS CTG definition
                    line1_id = CSL1.Fields("line_id")
                    line1Found = True
                    Print #File_CSB2B, "// Line1 ID#: " & line1_id
                ElseIf Not assessQueueDependency(CSL1.Fields("line_id"), incNoIA, ignoreAfterQueueNum) Then
                    Print #File_CSB2B, "// CTG blocked due to Line 1 (line_id=" & CSL1.Fields("line_id") & ") Queue / Not Queue logic"
                End If
                CSL1.MoveNext
            Wend
            'Cleanup
            CSL1.Close
            Set CSL1 = Nothing

            'Line 2 (repeat the same routine as line 1)
            On Error Resume Next
            db.QueryDefs.Delete "CSL2"
            On Error GoTo 0
            SQL = "SELECT * FROM Line INNER JOIN CommonStructureCkts ON Line.line_name = CommonStructureCkts.Line2Name WHERE (((CommonStructureCkts.ID)=" & CSCTG_id & "))"
            Set CSL2 = db.OpenRecordset(SQL, dbOpenSnapshot)
            CSL2.MoveLast
            CSL2.MoveFirst

            While Not CSL2.EOF And Not line2Found
                If CSL2.Fields("start_date") <= snapshotDate And snapshotDate < CSL2.Fields("end_date") And CSL2.Fields("InclAsCTG") And assessQueueDependency(CSL2.Fields("line_id"), incNoIA, ignoreAfterQueueNum) Then
                    line2_id = CSL2.Fields("line_id")
                    line2Found = True
                    Print #File_CSB2B, "// Line2 ID#: " & line2_id
                ElseIf Not assessQueueDependency(CSL2.Fields("line_id"), incNoIA, ignoreAfterQueueNum) Then
                    Print #File_CSB2B, "// CTG blocked due to Line 2 (line_id=" & CSL2.Fields("line_id") & ") Queue / Not Queue logic"
                End If
                CSL2.MoveNext
            Wend
            'Cleanup
            CSL2.Close
            Set CSL2 = Nothing

            '***INSERT BRANCH WRITING CODE HERE***
            'this code is similar to the code used when writing the combinable contingency file. It just writes valid branches for two line IDs instead of one
			
        'verify a valid line was found for both line1 and line 2
        If line1Found And line2Found Then
            foundData = False
            branchString = ""

            'line 1
            'create query for the first line's branches
            On Error Resume Next
            db.QueryDefs.Delete "branchFilter1"
            On Error GoTo 0
            SQL = "SELECT Branch.line_id, Branch.include, Branch.bus1, Branch.bus2, Branch.ckt, Branch.start_date, Branch.end_date FROM Branch WHERE (((Branch.line_id)=" & line1_id & "))"
            Set branchFilter1 = db.OpenRecordset(SQL, dbOpenSnapshot)
            branchFilter1.MoveLast
            branchFilter1.MoveFirst

            'determine which branches are valid and write them out to the contingency file
            While Not branchFilter1.EOF
                If branchFilter1.Fields("start_date") <= snapshotDate And snapshotDate <= branchFilter1.Fields("end_date") And branchFilter1.Fields("include") Then

                    If Not foundData Then
                        If branchString <> "" Then

Private Function assessQueueDependency(lineID As Integer, incWOIA As Boolean, ignoreThreshold As Integer) As Boolean
    'determine whether or not to write a B2B line contingency based on queue logic
    'default return is False (do not write line con) unless the following conditions are met
    assessQueueDependency = False

    'define and init variables
    Dim isQueue As Boolean
    Dim isNotQueue As Boolean
    Dim queueNum As Integer
	
    Dim notQueueNum As Integer
    queueNum = 0
    notQueueNum = 0
    Dim queuedProjectRecords As DAO.Recordset
    Dim lineRecords As DAO.Recordset
    Dim SQL As String
    isQueue = False
    isNotQueue = False

    'delete queries if they already exist
    On Error Resume Next
    db.QueryDefs.Delete "queuedProjectRecords"
    db.QueryDefs.Delete "lineRecords"
    On Error GoTo 0

    'create query that finds the line entered in the function parameters
    SQL = "SELECT Line.Queue, Line.NotQueue, Line.line_id FROM Line WHERE (((Line.line_id)=" & lineID & "))"
    Set lineRecords = db.OpenRecordset(SQL, dbOpenSnapshot)
    lineRecords.MoveLast
    lineRecords.MoveFirst

    'initial check to see if the line is dependent on queue/not queue logic at all
    If (IsNull(lineRecords.Fields("Queue"))) And IsNull(lineRecords.Fields("NotQueue")) Then
        assessQueueDependency = True
        'MsgBox "True Init"
        'cleanup
        lineRecords.Close
        Set lineRecords = Nothing
        Exit Function
    Else ' otherwise, determine queue numbers present
        If Not IsNull(lineRecords.Fields("Queue")) And Not IsNull(lineRecords.Fields("NotQueue")) Then
            queueNum = lineRecords.Fields("Queue")
            isQueue = True
            notQueueNum = lineRecords.Fields("NotQueue")
            isNotQueue = True
        Else
            If Not IsNull(lineRecords.Fields("Queue")) Then
                queueNum = lineRecords.Fields("Queue")
                isQueue = True
                'MsgBox "Queue #: " & queueNum
            End If
            If Not IsNull(lineRecords.Fields("NotQueue")) Then
                notQueueNum = lineRecords.Fields("NotQueue")
                isNotQueue = True
                'MsgBox "NotQueue #: " & notQueueNum
            End If
        End If
    End If

    'cleanup
    lineRecords.Close
    Set lineRecords = Nothing

    'criteria for determining a project is real and/or needs to be written out:
    '1. The include box is checked on the QueuedProjects page
    '2. The project has not withdrawn
    '3. The project has a signed IA, OR the "Include without IA" box is checked on the model build form
    '4. Queue num is less than or equal to the ignore after threshold
    'Note that valid start/end dates are not assessed here

    'NotQueue takes priority over Queue, so check that first
    If isNotQueue Then
        'create query for the "NotQueue" value
        SQL = "SELECT QueuedProjects.Include, QueuedProjects.QueueNum, QueuedProjects.StartDate, QueuedProjects.EndDate, QueuedProjects.SignedIA, QueuedProjects.[W/D] FROM QueuedProjects WHERE (((QueuedProjects.QueueNum)=" & notQueueNum & "))"
        Set queuedProjectRecords = db.OpenRecordset(SQL, dbOpenSnapshot)
        queuedProjectRecords.MoveLast
        queuedProjectRecords.MoveFirst
        If Not queuedProjectRecords.EOF Then ' there should be only one project with this queue numbers
            If (queuedProjectRecords.Fields("Include") And Not queuedProjectRecords.Fields("[W/D]")) And (queuedProjectRecords.Fields("SignedIA") Or incWOIA) And notQueueNum <= ignoreThreshold Then
                'the project with this queue number is valid, so it should block the line from being written as a con
                assessQueueDependency = False
                queuedProjectRecords.Close
                Set queuedProjectRecords = Nothing
                Exit Function
            End If
        'otherwise, if there is a queue number, we need to assess if that is a valid project. if yes, write the con for this line. If no, do not write the con for this line.
    End If
        End If

    If isQueue Then
        'create query for the "Queue" value
        SQL = "SELECT QueuedProjects.Include, QueuedProjects.QueueNum, QueuedProjects.StartDate, QueuedProjects.EndDate, QueuedProjects.SignedIA, QueuedProjects.[W/D] FROM QueuedProjects WHERE (((QueuedProjects.QueueNum)=" & queueNum & "))"
        Set queuedProjectRecords = db.OpenRecordset(SQL, dbOpenSnapshot)
        queuedProjectRecords.MoveLast
        queuedProjectRecords.MoveFirst
        If Not queuedProjectRecords.EOF Then ' there should be only one project with this queue numbers
            If (queuedProjectRecords.Fields("Include") And Not queuedProjectRecords.Fields("[W/D]")) And (queuedProjectRecords.Fields("SignedIA") Or incWOIA) And queueNum <= ignoreThreshold Then
                'the project with this queue number is valid and in range, so it should write the con for this line
                assessQueueDependency = True
                queuedProjectRecords.Close
                Set queuedProjectRecords = Nothing
                Exit Function
            End If
        End If
    ElseIf isNotQueue Then
        ' if we get here, there was an invalid NotQueue project, but no Queue project number. This means the project in NotQueue is not valid, and should not block the contingency
        assessQueueDependency = True
        queuedProjectRecords.Close
        Set queuedProjectRecords = Nothing
        Exit Function
    Else
        'If we get here, there was some kind of error because isQueue and isNotQueue were false
        MsgBox "Error"
    End If

    'if we make it here, the project assessed in the queue if statements above was not valid. Need to block the contingency from writing
    assessQueueDependency = False
    queuedProjectRecords.Close
    Set queuedProjectRecords = Nothing
    Exit Function
End Function

Public Sub WriteLatsLongsAxd(FilePath As String, SnapshotIndex As Integer, useRegionalNumbering As Boolean, busOffset As Long, combineBuses As Boolean)
    'Call RawWriter.WriteLatsLongsAxd(FilePath, Snapshot, Regional, BusOffset, CombineBuses)
    Dim SQL As String
	
    Dim File As Integer

    Dim BranchesLatsLongs As DAO.Recordset

    Set db = CurrentDb()

    Dim Snapshots As DAO.Recordset
    Dim SnapshotName As String
    Dim SnapshotYear As Long
    Dim SnapShotSeason As String
    Dim snapshotDate As Date

    Dim busFr As Long, busTo As Long, branchString As String, lineCount As Long

    'get snapshot info:
    Set Snapshots = db.OpenRecordset("Snapshots", dbOpenDynaset, dbSeeChanges)
    Snapshots.MoveLast
    Snapshots.MoveFirst
    While Not Snapshots.EOF
        If Snapshots.Fields("model_index") = SnapshotIndex Then
            SnapshotName = Snapshots.Fields("model_name")
            SnapshotYear = Snapshots.Fields("year")
            SnapShotSeason = Snapshots.Fields("season")
            snapshotDate = Snapshots.Fields("date")
        End If
        Snapshots.MoveNext
    Wend
    Snapshots.Close

    If Not useRegionalNumbering Then _
        busOffset = 0

    File = FreeFile
    If Dir$(FilePath & ".axd") <> "" Then _
        Kill (FilePath & ".axd")
    Open FilePath & ".axd" For Output As #File

    'print file metadata:
    Print #File, "// PowerWorld Geographical Branch File for SCEG Control Area"
    Print #File, "// Coordinates are Mercator"
    Print #File, "// Auto-Generated Snapshot from SCEGModelData DB"
    Print #File, "// File Created: " & Now()
    Print #File, "// Snapshot Year: " & SnapshotYear
    Print #File, "// Snapshot Season: " & SnapShotSeason
    Print #File, "// Snapshot Date: " & snapshotDate
    Print #File, "//--------------------------------------------------------------------------------"
    Print #File, "DATA (DisplayTransmissionLine, [BusNum,BusNum:1,LineCircuit,SOX,SOY,BranchDeviceType,SOThickness,SOColor,SOAnchored,SOLevel,SLName,SODashed])"
    Print #File, "{"

    lineCount = 0

    Set BranchesLatsLongs = db.OpenRecordset("BranchesLats&Longs", dbOpenDynaset, dbSeeChanges)

    While Not BranchesLatsLongs.EOF
	
        If BranchesLatsLongs.Fields("start_date") <= snapshotDate And snapshotDate <= BranchesLatsLongs.Fields("end_date") And BranchesLatsLongs.Fields("include") Then
            If BranchesLatsLongs.Fields("bus1") < 9999 Then
                busFr = BranchesLatsLongs.Fields("bus1") + busOffset
            Else
                busFr = BranchesLatsLongs.Fields("bus1")
            End If

            If BranchesLatsLongs.Fields("bus2") < 9999 Then
                busTo = BranchesLatsLongs.Fields("bus2") + busOffset
            Else
                busTo = BranchesLatsLongs.Fields("bus2")
            End If

            If BranchesLatsLongs.Fields("StartPoint") = -1 Then ' True
                If lineCount > 0 Then _
                    Print #File, "   </SUBDATA>"

                branchString = " " & busFr & "  " & busTo & " """ & BranchesLatsLongs.Fields("ckt") & """ " & BranchesLatsLongs.Fields("Lat") & " " & BranchesLatsLongs.Fields("Long") & " ""Line"" 1 "

                If BranchesLatsLongs.Fields("base_kV") < 32 Then
                    branchString = branchString & "$00008000 "   ' Green
                ElseIf BranchesLatsLongs.Fields("base_kV") < 45 Then
                    branchString = branchString & "$0000FFFF "     ' Yellow
                ElseIf BranchesLatsLongs.Fields("base_kV") < 114 Then
                    branchString = branchString & "$FFFFFFFF "     ' Blue
                ElseIf BranchesLatsLongs.Fields("base_kV") < 229 Then
                    branchString = branchString & "$000000FF "     ' Red
                ElseIf BranchesLatsLongs.Fields("base_kV") < 344 Then
                    branchString = branchString & "$00FFFFFF "     ' White
                Else
                    branchString = branchString & "$000080FF "   ' Orange
                End If

                branchString = branchString & """YES"" ""Top"" """ & BranchesLatsLongs.Fields("base_kV") & "kV"""

                If (BranchesLatsLongs.Fields("bus1") < 9999 And BranchesLatsLongs.Fields("bus2") < 9999) Or (BranchesLatsLongs.Fields("bus1") > 9999 And BranchesLatsLongs.Fields("bus2") > 9999) Then ' Branch is uni-owned
                    branchString = branchString & " ""     0"" " ' Solid line
                Else ' Branch is multi-owned
                    branchString = branchString & " ""    3"" " ' Dashed line
                End If

                Print #File, branchString
                Print #File, "    <SUBDATA Line>"

            End If

            branchString = "   [" & BranchesLatsLongs.Fields("Lat") & " " & BranchesLatsLongs.Fields("Long") & "]"
            If Not IsNull(BranchesLatsLongs.Fields("Lat1")) And BranchesLatsLongs.Fields("Lat1") <> "" Then _
                branchString = branchString & " [" & BranchesLatsLongs.Fields("Lat1") & " " & BranchesLatsLongs.Fields("Long1") & "]"
                Print #File, branchString

            lineCount = lineCount + 1

        End If

        BranchesLatsLongs.MoveNext

    Wend
	
    BranchesLatsLongs.Close

    Print #File, "    </SUBDATA>"
    Print #File, "}" & vbCrLf
    Print #File, "SCRIPT"
    Print #File, "{"
    Print #File, "AutoInsertLineFlowPieCharts(0);"
    Print #File, "}"
    Close #File

End Sub
