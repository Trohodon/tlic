Function ParseBus(pValue, busOne As Boolean) As Long
    Dim ColonPosition As Integer, ParseLeftBus As String, ParseRightBus As String

    'Find position of colon
    ColonPosition = InStr(pValue, ":")

    'Return the portion of the string before the colon
    If ColonPosition > 0 Then
        ParseLeftBus = Left(pValue, ColonPosition - 1)
    Else
        ParseLeftBus = ""
    End If

    Dim LPosition As Integer

    'Find position of >
    LPosition = InStr(pValue, ">")

    'Return the portion of the string after the >
    If LPosition > 0 Then
        ParseRightBus = Mid(pValue, LPosition + 1)

        'Find position of colon
        LPosition = InStr(ParseRightBus, ":")

        'Return the portion of the string before the :
        If LPosition > 0 Then
            ParseRightBus = Left(ParseRightBus, LPosition - 1)
        Else
            ParseRightBus = ""
        End If
    Else
        ParseRightBus = ""
    End If

    If ParseLeftBus = "" Or ParseRightBus = "" Then
        ParseBus = 0
    Else
        If CLng(ParseLeftBus) < CLng(ParseRightBus) Then
            If busOne Then
                ParseBus = ParseLeftBus
            Else
                ParseBus = ParseRightBus
            End If
        Else
            If Not busOne Then
                ParseBus = ParseLeftBus
            Else
                ParseBus = ParseRightBus
            End If
        End If
    End If
End Function
