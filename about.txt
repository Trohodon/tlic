Public Function ExportToDyre() As Integer
    Dim Dir As String
    Dim Filename As String
    Dim snapshotDate As Date
    Dim Regional As Boolean
    Dim ExportAllModels As Boolean
    Dim IncludeFuture As Boolean
    Dim Pumping As Boolean
    Dim ConstantsPerRow As Integer
    Dim IncludeNoIA_Dyn As Boolean 'variable for 'Include Without IA' checkbox on the DynamicExport Form

    'Get export options from form:
    Form_DynamicExportForm.Text1.SetFocus
    Dir = Form_DynamicExportForm.Text1.Text

    Form_DynamicExportForm.Text16.SetFocus
    Filename = Form_DynamicExportForm.Text16.Text

    Form_DynamicExportForm.Text4.SetFocus
    snapshotDate = Form_DynamicExportForm.Text4.Value

    Form_DynamicExportForm.Check8.SetFocus
    ExportAllModels = Form_DynamicExportForm.Check8.Value

    Form_DynamicExportForm.CheckIncFuture.SetFocus
    IncludeFuture = Form_DynamicExportForm.CheckIncFuture.Value

    Form_DynamicExportForm.CheckIncFuture.SetFocus
    Pumping = Form_DynamicExportForm.CheckPumping.Value

    Form_DynamicExportForm.Check12.SetFocus
    Regional = Form_DynamicExportForm.Check12.Value

    Form_DynamicExportForm.Text14.SetFocus
    ConstantsPerRow = Form_DynamicExportForm.Text14.Value

    Form_DynamicExportForm.CheckNoIA_Dyn.SetFocus
    IncludeNoIA_Dyn = Form_DynamicExportForm.CheckNoIA_Dyn.Value

    'Call write to dyre function with options:
    WriteSnapshotToDyre ExportAllModels, IncludeFuture, Pumping, snapshotDate, Dir & "\" & Filename, Regional, ConstantsPerRow, IncludeNoIA_Dyn
End Function

Public Sub WriteSnapshotToDyre(ExportAll As Boolean, IncludeFuture As Boolean, Pumping As Boolean, snapshotDate As Date, Path As String, Reduced As Boolean, ConstPerRow As Integer, IncludeNoIA_Dyn As Boolean)
    Dim db As DAO.Database
    Dim Models As DAO.Recordset
    Dim QDF As DAO.QueryDef
    Dim Constants As DAO.Recordset

    Dim DyreFileNumber As Integer
    Dim ModelID As Long
    Dim bus As Long
    Dim BusName As String
    Dim BaseKV As Double

    Dim Queue As Variant
    Dim ModelType, MachID As String
    Dim StartDate, EndDate, ModelDateStart, ModelDateEnd As Date
    Dim SignedIA As Variant
    Dim Line, i As String
    Dim Field As String

    'TURN OFF WARNINGS!
    DoCmd.SetWarnings False
    Set db = CurrentDb()
    DyreFileNumber = FreeFile
    Open Path For Output As #DyreFileNumber
    Set Models = db.OpenRecordset("DynamicsModelInfo", dbOpenDynaset, dbSeeChanges)
    Models.MoveLast
    Models.MoveFirst
    LineOld = ""

    While Not Models.EOF
        If Models.Fields("include").Value Then
            Line = ""
            i = 2
            ModelID = Models.Fields("model_id").Value
            BusName = Models.Fields("reduced_name").Value
            bus = Models.Fields("bus_number").Value

            If Reduced Then
                bus = bus + 370000
            End If

            BaseKV = Models.Fields("base_kV").Value
            ModelType = Models.Fields("model_type_name").Value
            MachID = Models.Fields("pti_machine_id").Value
            StartDate = Models.Fields("start_date").Value
            EndDate = Models.Fields("end_date").Value
            ModelDateStart = Models.Fields("model_date_start").Value
            ModelDateEnd = Models.Fields("model_date_end").Value
            Queue = Models.Fields("Queue").Value
            SignedIA = Models.Fields("SignedIA").Value

            If IncludeNoIA_Dyn Or IsNull(Models.Fields("Queue")) Or (Not IsNull(Models.Fields("Queue")) And SignedIA = -1) Then
                If (((snapshotDate >= StartDate Or IncludeFuture) And snapshotDate < EndDate And (snapshotDate >= ModelDateStart Or IncludeFuture) And snapshotDate < ModelDateEnd) Or ExportAll) _
                    And Not (Pumping And Models.Fields("bus_number") > 820 And Models.Fields("bus_number") < 829 And (ModelType Like "IEEEG2" Or ModelType Like "WSHYDD")) Then

                    Line = Line + FormatLong(bus, 8)

                    If Models.Fields("is_user_model") Then
                        Line = Line & "USRMDL" & " " & FormatStr(MachID, 13) & FormatStr("'" & ModelType & "'", 13)
                        i = i + 1 ' Need one less column since USRMDL printout takes up extra column
                    Else
                        Line = Line & FormatStr("'" & ModelType & "'", 10) & FormatStr(MachID, 13)
                    End If

                    Set QDF = db.QueryDefs("DynamicsConstantsByModelID")
                    QDF.Parameters("ModelID") = ModelID
                    Set Constants = QDF.OpenRecordset

                    If Not Constants.EOF Then
                        Constants.MoveLast
                        Constants.MoveFirst

                        While Not Constants.EOF
                            Field = FormatDbl(Constants.Fields("const_value").Value, 13, 8)

                            If Constants.Fields("is_ICON").Value Then
                                Field = FormatDbl(Constants.Fields("const_value").Value, 13, 0)
                            End If

                            If Constants.Fields("is_bus_number").Value And Reduced Then
                                If Constants.Fields("const_value").Value > 0 Then
                                    Field = FormatDbl(370000 + Constants.Fields("const_value").Value, 13, 0)
                                Else
                                    Field = FormatDbl(0, 13, 0)
                                End If
                            End If

                            Line = Line & Field
                            Constants.MoveNext
                            i = i + 1

                            If i > ConstPerRow Then
                                If Not Constants.EOF Then
                                    Print #DyreFileNumber, Line
                                    Line = Space(18)
                                    i = 1
                                End If
                            End If
                        Wend
                        Else
                            MsgBox "Blank values exist in the ""DynamicsConstantsByModelID"" query! Not good."
                        End If

                        While i <= ConstPerRow
                            Line = Line & Space(13)
                            i = i + 1
                        Wend

                        Line = Line & " / " & FormatStr(BusName, 8) & " " & Format(BaseKV, "0.0")
                        Line = Line & "   " & Format(ModelDateStart, "mmm-dd-yyyy") & ", " & Format(ModelDateEnd, "mmm-dd-yyyy")
                        Print #DyreFileNumber, Line
                    End If
                End If ' ^If (((snapshotDate >= StartDate Or IncludeFuture) And snapshotDate < EndDate And (snapshotDate >= ModelDateStart Or IncludeFuture) And snapshotDate < ModelDateEnd)
            End If ' ^If IncludeNoIA_Dyn Or IsNull(Models.Fields("Queue")) Or (Not IsNull(Models.Fields("Queue")) And SignedIA = -1) Then
        End If ' ^If Models.Fields("include").Value Then

        Models.MoveNext
    Wend ' ^While Not Models.EOF

    Close #DyreFileNumber

    'TURN ON WARNINGS!
    DoCmd.SetWarnings True

    MsgBox "Done"

End Sub

Public Function FormatDbl(num As Double, Width As Integer, Precision As Integer)

    Dim DecFormat As String
    Dim i As Integer

    If Precision > 0 Then
        DecFormat = "0.0"
    Else
        DecFormat = "0"
    End If

    If num = 0# Then
        FormatDbl = Format(num, DecFormat) & Space(Width - Len(Format(num, DecFormat)))
    ElseIf Round(num, 0) = num Then
        FormatDbl = Format(num, DecFormat) & Space(Width - Len(Format(num, DecFormat)))
    Else
        FormatDbl = Format(num) & Space(Width - Len(Format(num)))
    End If

End Function

Public Function FormatInt(num As Integer, Width As Integer)

    FormatInt = num & Space(Width - Len(Format(num, "0")))

End Function

Public Function FormatLong(num As Long, Width As Integer)

    FormatLong = num & Space(Width - Len(Format(num, "0")))

End Function

Public Function FormatStr(str As String, Width As Integer)
    If (Width - Len(str)) >= 0 Then
        FormatStr = str & Space(Width - Len(str))
    Else
        FormatStr = str
    End If
End Function
