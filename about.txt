Public Function returnLoadValue(snapDate As Date, snapSeason As String, SnapType As String) As Integer
    returnLoadValue = 0

    Dim rec As Recordset
    Dim found As Boolean
    found = False

    Set db = CurrentDb
    Set rec = db.OpenRecordset("Corporate Load Forecast", dbOpenDynaset, dbSeeChanges)

    rec.MoveLast
    rec.MoveFirst

    While Not rec.EOF
        If rec.Fields("Year") = Year(snapDate) And rec.Fields("Month") = month(snapDate) And Not found Then
            found = True

            ' Defaults to the 50/50 forcast(Gross Peak)
            Dim PeakType As String
            PeakType = "Gross Peak"

            If SnapType = "TPL" Or SnapType = "VAR" Or SnapType = "GEN CNXN" Or SnapType = "90/10" Then
                PeakType = "90/10"
            ElseIf SnapType = "High Peak" Then
                PeakType = "High Peak"
            End If

            rec.MovePrevious
            On Error Resume Next
            If (snapSeason <> "Z" And snapSeason <> "F") Or (snapSeason = "Z" And month(snapDate) <> 3) Or (snapSeason = "F" And month(snapDate) <> 9) Then _
                returnLoadValue = Round(rec.Fields(PeakType), 0)
            rec.MoveNext
            If Not rec.EOF And returnLoadValue < Round(rec.Fields(PeakType), 0) Then _
                returnLoadValue = Round(rec.Fields(PeakType), 0)
                If Not rec.EOF Then
                    rec.MoveNext
                    If Not rec.EOF Then
                        If returnLoadValue < Round(rec.Fields(PeakType), 0) And ((snapSeason <> "Z" And snapSeason <> "F") Or (snapSeason = "Z" And month(snapDate) <> 5) Or (snapSeason = "F" And month(snapDate) <> 11)) Then _
                            returnLoadValue = Round(rec.Fields(PeakType), 0)
                End If
            End If

            On Error GoTo -1

            If snapSeason = "L" Then
                'returnLoadValue = returnLoadValue * 0.342
                returnLoadValue = 1756 ' 1 percentile of valley for the last 12 years. JWR - 2017-05-26
            ElseIf snapSeason = "H" Then
                returnLoadValue = returnLoadValue * 0.8
            ElseIf Left(snapSeason, 3) = "DTM" Then
                'returnLoadValue = returnLoadValue * 0.584
                If Right(snapSeason, 1) = "g" Then
                    returnLoadValue = 2138 ' 1 percentile of min DTM for the last 12 years. JWR - 2017-05-26
                ElseIf Right(snapSeason, 1) = "Z" Then
                    returnLoadValue = 2164 ' 1 percentile of min spring DTM for the last 4 years. JWR - 2017-12-22
                ElseIf Right(snapSeason, 1) = "S" Then
                    returnLoadValue = 2639 ' 1 percentile of min summer DTM for the last 4 years. JWR - 2017-12-22
                ElseIf Right(snapSeason, 1) = "F" Then
                    returnLoadValue = 2197 ' 1 percentile of min fall DTM for the last 4 years. JWR - 2017-12-22
                Else
                    returnLoadValue = 2162 ' 1 percentile of min winter DTM for the last 4 years. JWR - 2017-12-22
                End If
            End If
        End If

        rec.MoveNext
    Wend

    rec.Close

    Set rec = Nothing
    Set db = Nothing
End Function

Public Function returnInterchangeValue(snapDate As Date, snapSeason As String) As Integer
    returnInterchangeValue = 0

    Dim rec As Recordset

    Set db = CurrentDb
    Set rec = db.OpenRecordset("Schedules(Interchange)", dbOpenDynaset, dbSeeChanges)

    rec.MoveLast
    rec.MoveFirst

    While Not rec.EOF
        If rec.Fields("Start") <= snapDate And rec.Fields("End") >= snapDate Then
            If snapSeason <> "L" Then
                returnInterchangeValue = returnInterchangeValue + rec.Fields("MW")
            ElseIf Not rec.Fields("Peak_only") Then
                returnInterchangeValue = returnInterchangeValue + rec.Fields("MW")
            End If
        End If

        rec.MoveNext
    Wend

    rec.Close

    Set rec = Nothing
    Set db = Nothing
End Function

Public Function returnPercGenMax(snapSeason As String, SnapType As String) As Integer
    returnPercGenMax = 0

    If SnapType = "GEN CNXN" And snapSeason <> "W" And snapSeason <> "L" Then 'Added And snapSeason <> "W" and "L" 9/1/20 for Battery Studies AU
        returnPercGenMax = 100
    ElseIf snapSeason = "S" Then
        returnPercGenMax = 81
    ElseIf snapSeason = "Z" Or snapSeason = "F" Then
        returnPercGenMax = 83
    ElseIf snapSeason = "W" Or snapSeason = "L" Then 'Added Or snapSeason = "L" AU 9/3/20
        returnPercGenMax = 0
    ElseIf snapSeason <> "W" And snapSeason <> "L" Then
        returnPercGenMax = 100
    End If
End Function

Public Function setSolarSysLd(snapSeason As String, SnapLoad As Integer, SnapType As String, snapDate As Date, baseYear As Double, incSolar As Boolean, percentOfSolar As Integer, ignoreAfterQueueNum As Integer, _
                                incNoIA As Boolean, incAffSys As Boolean, dispDTM As Integer) As Double

    Dim Generators As DAO.Recordset, QueuedProjects As DAO.Recordset
    Dim TotalGen As Double, Available As Boolean, StartDate As Date, EndDate As Date, Queue As Double, SignedIA As Boolean, Pmax As Double, Include As Boolean

    TotalGen = 0

    Set db = CurrentDb()

    Set QueuedProjects = db.OpenRecordset("QueuedProjects", dbOpenDynaset, dbSeeChanges)
    Set Generators = db.OpenRecordset("MachinesSorted", dbOpenDynaset, dbSeeChanges)
    Generators.MoveLast
    Generators.MoveFirst

    While Not Generators.EOF
        If IsNumeric(Generators.Fields("Queue").Value) Then
            Include = Generators.Fields("include").Value
            Available = Generators.Fields("available").Value
            StartDate = Generators.Fields("start_date").Value
            EndDate = Generators.Fields("end_date").Value
            Queue = Generators.Fields("Queue").Value

            QueuedProjects.MoveLast
            QueuedProjects.MoveFirst
            While Not QueuedProjects.EOF And Queue <> QueuedProjects.Fields("QueueNum")
                QueuedProjects.MoveNext
            Wend
            SignedIA = QueuedProjects.Fields("signedIA").Value

            If Right(snapSeason, 1) <> "W" Then
                Pmax = Generators.Fields("pmax_MW").Value
            Else
                Pmax = Generators.Fields("pmax_MW_win").Value
            End If

            If Include And Available And snapDate >= StartDate And snapDate < EndDate And Queue <= ignoreAfterQueueNum And (incNoIA Or (Not incNoIA And SignedIA)) Then
                TotalGen = TotalGen * percentOfSolar / 100 + Pmax
            End If
        End If
        Generators.MoveNext
    Wend

    Generators.Close

    setSolarSysLd = TotalGen + dispDTM

    If SnapLoad > setSolarSysLd Then
        setSolarSysLd = SnapLoad
    End If

End Function
