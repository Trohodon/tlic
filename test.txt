Public Sub CalculateAllRatings() 'Recalc Ratings Button
    ' Dim line As TLICs.TLine 'Init Line object. This object is contained in a COM-wrapped .NET assembly
    ' Set ElectricalLineModel = New TLineCalc.TLine 'instantiate Line object
    ' Set TLineCalc = New TLineLib.TLine 'instantiate Line object

    Dim db As DAO.Database 'Database handle
    Set db = CurrentDb 'Get handle to current database
    Dim Result As VbMsgBoxResult 'Confirm execute msg box result
    Result = vbCancel
    Dim SQL As String, calcRatingsString As String

    ' Calc BranchSection ratings
    Dim Sections As DAO.Recordset 'Recordset object for Section Table
    Dim SectionID As Long 'Section id field cache
    Dim SectionName As String 'Section name field cache
    Set Sections = db.OpenRecordset("BranchSections") 'Open BranchSections Query

    'Reset row cursor
    Sections.MoveLast
    Sections.MoveFirst

    If Not (Sections.EOF) Then 'If table not empty then:
        Result = MsgBox("This will take a few minutes either way, but would you like to clear all values then recalculate (will take much longer)?", vbYesNoCancel, "RatingsCalc Routine") 'Warn user of time requirements
        If Result = vbYes Or Result = vbNo Then 'If user is OK with time then:
            DoCmd.SetWarnings (False) 'Turn off warnings
            'Loop through sections, calculate section ratings, and write ratings back to section record:
            While Not Sections.EOF 'Until the end of the table:
                SectionID = Sections.Fields("section_id").Value 'Get the section id from the current row
                SectionName = Sections.Fields("section_name").Value 'Get the section name from the current row
                If IsNumeric(Left(SectionName, 1)) Or UCase(SectionName) = "NONAME" Or UCase(Left(SectionName, 2)) = "CD" _
                    Or UCase(Left(SectionName, 3)) = "BUS" Or UCase(Left(SectionName, 4)) = "FOLD" Then ' Only calc ratings for conductors (their sect name starts with number, NONAME, CD (Ckt breaker Drop)), BUS or FOLD (Fold-in)
                    tempF = Sections.Fields("MOT_°F").Value
                    phase_cond_ID = Sections.Fields("phase_cond_id").Value
                    RebuiltSoon = Sections.Fields("RebuiltSoon").Value
                    If IsNumeric(Sections.Fields("BundledSpacing").Value) Then ' If num value present assign it
                        BundledSpacing = Sections.Fields("BundledSpacing").Value
                    Else ' Otherwise set it to 12
                        BundledSpacing = 12
                    End If

                    calcRatingsString = CalculateSectionRatings(False)
                    If calcRatingsString <> "" Then
                        'Build SQL string to write ratings to section table:
                        SQL = "UPDATE BranchSections SET " & calcRatingsString & " WHERE section_id = " & SectionID

                        DoCmd.RunSQL SQL 'Run SQL
                    End If
                ElseIf Result = vbYes And UCase(SectionName) <> "NEIGHBOR" Then
                    'Build SQL string to clear out ratings in section table:
                    'SQL = "UPDATE BranchSections SET rate_A = """", rate_B = """", rate_C = """", rate_A_win = """", rate_B_win = """", rate_C_win = """" WHERE section_id = " & SectionID
                    'Removed line above and added below. 10/2023 BU
                    SQL = "UPDATE BranchSections SET rate_A = """", rate_B = """", rate_C = """", rate_A_win = """", rate_B_win = """", rate_C_win = """"," _
					& "rate_A_WinPeak = """", rate_B_WinPeak = """", rate_C_WinPeak = """", rate_A_OpsPlnWin = """", rate_B_OpsPlnWin = """", rate_C_OpsPlnWin = """"," _
                    & "rate_A_Spr = """", rate_B_Spr = """", rate_C_Spr = """", rate_A_Sum = """", rate_B_Sum = """", rate_C_Sum = """"," _
					& "rate_A_Fall = """", rate_B_Fall = """", rate_C_Fall = """" WHERE section_id = " & SectionID

                    DoCmd.RunSQL SQL
                End If

                Sections.MoveNext 'Move cursor to next row
            Wend
            'Print errors to user:
            If SomeNotFound Then 'If one or more sections did not compute then:
                MsgBox "These sections did not have their ratings calculated:" & vbCrLf & NotFound 'print errors
            ' Else
            '     MsgBox "Done with no errors."
            End If
        End If
    ElseIf Sections.EOF Then
        MsgBox "The BranchSections table is empty. Quitting!"
    Else
        MsgBox "The Conductors table is empty. Quitting!"
    End If

    If Result = vbYes Or Result = vbNo Then
        ' Calc TwoWindingsSections ratings
        Dim TwoWSections As DAO.Recordset 'Recordset object for Section Table
        Dim TwoWSectionID As Long 'TwoWSection id field cache
        Dim TwoWSectionName As String 'TwoWSection name field cache
        Set TwoWSections = db.OpenRecordset("TwoWindingsSections") 'Open TwoWindingsSections Query

        'Reset row cursor
        TwoWSections.MoveLast
        TwoWSections.MoveFirst

        If Not (TwoWSections.EOF) Then 'If table not empty then:
            'Loop through sections, calculate section ratings, and write ratings back to section record:
            While Not TwoWSections.EOF 'Until the end of the table:
                TwoWSectionID = TwoWSections.Fields("section_id").Value 'Get the section id from the current row
                TwoWSectionName = TwoWSections.Fields("section_name").Value 'Get the section name from the current row
                If IsNumeric(Left(TwoWSectionName, 1)) Or UCase(TwoWSectionName) = "NONAME" Or UCase(Left(TwoWSectionName, 2)) = "CD" _
                    Or UCase(Left(TwoWSectionName, 3)) = "BUS" Then ' Only calc ratings for conductors (their sect name starts with number, NONAME, CD (Ckt breaker Drop)) OR BUS
                    tempF = 500 ' Setting to 500 to make sure MOT is always highest possible for TwoWindings.
                    phase_cond_ID = TwoWSections.Fields("phase_cond_id").Value
                    RebuiltSoon = False
                    If IsNumeric(TwoWSections.Fields("BundledSpacing").Value) Then ' If num value present assign it
                        BundledSpacing = TwoWSections.Fields("BundledSpacing").Value
                    Else ' Otherwise set it to 12
                        BundledSpacing = 12
                    End If

                    calcRatingsString = CalculateSectionRatings(False)
                    If calcRatingsString <> "" Then
                        'Build SQL string to write ratings to section table:
                        SQL = "UPDATE TwoWindingsSections SET " & calcRatingsString & " WHERE section_id = " & TwoWSectionID

                        DoCmd.RunSQL SQL 'Run SQL
                    End If
                ElseIf Result = vbYes Then
                    'Build SQL string to clear out ratings in TwoWindingsSection table:
                    ' SQL = "UPDATE TwoWindingsSections SET rate_A = """", rate_B = """", rate_C = """", rate_A_win = """", rate_B_win = """", rate_C_win = """" WHERE section_id = " & TwoWSectionID
					'
                    'SQL = "UPDATE TwoWindingsSections SET Amp_rate_A = """", Amp_rate_B = """", Amp_rate_C = """", Amp_rate_A_win = """", Amp_rate_B_win = """", Amp_rate_C_win = """" WHERE section_id = " & TwoWSectionID
                    'Removed line above and added below. 10/2023 BU
                    SQL = "UPDATE TwoWindingsSections SET Amp_rate_A = """", Amp_rate_B = """", Amp_rate_C = """", Amp_rate_A_win = """", Amp_rate_B_win = """", Amp_rate_C_win = """"," _ 
					& "Amp_rate_A_WinPeak = """", Amp_rate_B_WinPeak = """", Amp_rate_C_WinPeak = """", Amp_rate_A_OpsPlnWin = """", Amp_rate_B_OpsPlnWin = """", Amp_rate_C_OpsPlnWin = """"," _
                    & "Amp_rate_A_Spr = """", Amp_rate_B_Spr = """", Amp_rate_C_Spr = """", Amp_rate_A_Sum = """", Amp_rate_B_Sum = """", Amp_rate_C_Sum = """"," _
					& "Amp_rate_A_Fall = """", Amp_rate_B_Fall = """", Amp_rate_C_Fall = """" WHERE section_id = " & TwoWSectionID

                    DoCmd.RunSQL SQL
                End If

                TwoWSections.MoveNext 'Move cursor to next row
            Wend
            'Print errors to user:
            If SomeNotFound Then 'If one or more sections did not compute then:
                MsgBox "These TwoWindings sections did not have their ratings calculated:" & vbCrLf & NotFound 'print errors
            Else
                MsgBox "Done with no errors."
            End If

            DoCmd.SetWarnings True 'turn warnings back on
        ElseIf TwoWSections.EOF Then
            MsgBox "The TwoWindingsSections table is empty. Quitting!"
        Else
            MsgBox "The Conductors table is empty. Quitting!"
        End If

        'now update the 'TimeStamp' table so the time stamp text box on the ModelBuildForm is updated below each button.
        DoCmd.SetWarnings False
        DoCmd.RunSQL "UPDATE TimeStamp1 SET TimeStamp1.Recalc_Ratings = " & Chr(39) & Format(Now(), "MM/dd/yyyy  hh:mm AM/PM") & Chr(39) & " WHERE TimeStamp_ID = " & 1 & ";"
        DoCmd.SetWarnings True
    End If
End Sub

Public Function CalculateSectionRatings(Verbose As Boolean) As String
    CalculateSectionRatings = ""

    Dim db As DAO.Database 'Database handle
    Set db = CurrentDb 'Get handle to current database

    Dim Conds As DAO.Recordset 'Recordset object for Conductors Table
    Set Conds = db.OpenRecordset("Conductors")

    Dim diam_mm As Double, diam_in As Double, rLow As Double, rHigh As Double ' Values to obtain from Conductors table
    Dim condName As String, condCodeName As String, tempC As Double, tempC_A As Double, tempC_B As Double, tempC_C As Double
    Dim RatingA As Double, RatingB As Double, RatingC As Double, WinRatingA As Double, WinRatingB As Double, WinRatingC As Double
    Dim WinPeakRatingA As Double, WinPeakRatingB As Double, WinPeakRatingC As Double, OpsPlnWinRatingA As Double, OpsPlnWinRatingB As Double, OpsPlnWinRatingC As Double    'added 10/2023 BU
    Dim SprRatingA As Double, SprRatingB As Double, SprRatingC As Double, SumRatingA As Double, SumRatingB As Double, SumRatingC As Double, FallRatingA As Double, FallRatingB As Double, FallRatingC As Double    'added 10/2023 BU
    tempC = (tempF - 32) * 5 / 9
    tempC_A = 94
    tempC_B = 100
    tempC_C = 125

    ' loop to find values for this conductor in Conductors table
    Conds.MoveLast
    Conds.MoveFirst
    Do While Not Conds.EOF 'Until the end of the table:
        If phase_cond_ID = Conds.Fields("id").Value And IsNumeric(Conds.Fields("OD_IN").Value) And IsNumeric(Conds.Fields("R25").Value) _
        And IsNumeric(Conds.Fields("R75").Value) Then
            condName = Conds.Fields("cond").Value
            condCodeName = Conds.Fields("CODE_NAME").Value
            diam_in = Conds.Fields("OD_IN").Value
            diam_mm = Conds.Fields("OD_IN").Value / 0.03937 ' Convert in to mm
            rLow = Conds.Fields("R25").Value
            rHigh = Conds.Fields("R75").Value

            Exit Do
        End If

        If phase_cond_ID = Conds.Fields("id").Value And Left(Conds.Fields("cond"), 2) = "B " And BundledSpacing < 12 Then
            condName = Conds.Fields("cond").Value
            condCodeName = Conds.Fields("CODE_NAME").Value
            ' Set ratings all equal to values in the Conductors table - cable conductor ratings will be overwritten, but rigid bus will not.
            ' Note: have to half now, b/c bundlefactor will multiply later in the code.
            RatingA = Conds.Fields("rate_a") / 2
            RatingB = Conds.Fields("rate_b") / 2
            RatingC = Conds.Fields("rate_c") / 2
            WinRatingA = Conds.Fields("rate_a_win") / 2
            WinRatingB = Conds.Fields("rate_b_win") / 2
            WinRatingC = Conds.Fields("rate_c_win") / 2
            WinPeakRatingA = Conds.Fields("rate_a_WinPeak") / 2 'added 10/2023 BU
            WinPeakRatingB = Conds.Fields("rate_b_WinPeak") / 2 'added 10/2023 BU
            WinPeakRatingC = Conds.Fields("rate_c_WinPeak") / 2 'added 10/2023 BU
            OpsPlnWinRatingA = Conds.Fields("rate_a_OpsPlnWin") / 2 'added 10/2023 BU
            OpsPlnWinRatingB = Conds.Fields("rate_b_OpsPlnWin") / 2 'added 10/2023 BU
            OpsPlnWinRatingC = Conds.Fields("rate_c_OpsPlnWin") / 2 'added 10/2023 BU
            SprRatingA = Conds.Fields("rate_a_Spr") / 2 'added 10/2023 BU
            SprRatingB = Conds.Fields("rate_b_Spr") / 2 'added 10/2023 BU
            SprRatingC = Conds.Fields("rate_c_Spr") / 2 'added 10/2023 BU
            SumRatingA = Conds.Fields("rate_a_Sum") / 2 'added 10/2023 BU
            SumRatingB = Conds.Fields("rate_b_Sum") / 2 'added 10/2023 BU
            SumRatingC = Conds.Fields("rate_c_Sum") / 2 'added 10/2023 BU
            FallRatingA = Conds.Fields("rate_a_Fall") / 2 'added 10/2023 BU
            FallRatingB = Conds.Fields("rate_b_Fall") / 2 'added 10/2023 BU
            FallRatingC = Conds.Fields("rate_c_Fall") / 2 'added 10/2023 BU
            Exit Do
        End If

        Conds.MoveNext 'Move cursor to next row
    Loop

    If condName <> "" And InStr(condName, "TUBE") = 0 And InStr(condName, "Al Sch") = 0 And InStr(condName, "Al Rect") = 0 Then ' Conductor exists
        Dim condType As String
        condType = ""
        If InStr(condName, "-HY") > 0 Then ' Hytherm..must identify so can use higher MOT
            condType = "HY"
            tempC_C = tempC
            If tempC_A + 25 <= tempC Then
                tempC_A = tempC_A + 26
            Else
                tempC_A = tempC
            End If
            If tempC_B + 25 <= tempC Then
                tempC_B = tempC_B + 25
            Else
                tempC_B = tempC
            End If
        ElseIf Right(condName, 4) = "ACCC" Then ' ACCC..must identify so can use higher MOT
            condType = "ACCC"
            tempC_C = tempC
            If 180 <= tempC Then
                tempC_A = 180
            Else
                tempC_A = tempC
            End If
            If 200 <= tempC Then
                tempC_B = 200
            Else
                tempC_B = tempC
            End If
        ElseIf Right(condName, 4) = "ACCR" Then ' ACCR..must identify so can use higher MOT
            condType = "ACCR"
            tempC_C = tempC
            If 210 <= tempC Then
                tempC_A = 210
            Else
                tempC_A = tempC
            End If
            If 240 <= tempC Then
                tempC_B = 240
            Else
                tempC_B = tempC
            End If
        End If

                tempC_A = tempC
                tempC_B = tempC
                tempC_C = tempC
            ElseIf tempC < tempC_C Then
                tempC_A = tempC_B
                tempC_B = tempC
                tempC_C = tempC
            Else
                tempC_A = tempC_B
                tempC_B = tempC_C
                tempC_C = tempC
            End If

				35.6, 45#, 10#, 1#, 12#, True)
				
		Else
			RatingA = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_A, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				40#, 45#, 7#, 1#, 12#, True)
			RatingB = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_B, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				40#, 45#, 7#, 1#, 12#, True)
			RatingC = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_C, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				40#, 45#, 7#, 1#, 12#, True)

			WinRatingA = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_A, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				26.7, 45#, 3#, 1#, 12#, True)
			WinRatingB = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_B, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				26.7, 45#, 3#, 1#, 12#, True)
			WinRatingC = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_C, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				26.7, 45#, 3#, 1#, 12#, True)

			WinPeakRatingA = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_A, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				-2.8, 45#, 1#, 1#, 12#, True)
			WinPeakRatingB = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_B, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				-2.8, 45#, 1#, 1#, 12#, True)
			WinPeakRatingC = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_C, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				-2.8, 45#, 1#, 1#, 12#, True)

			OpsPlnWinRatingA = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_A, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				28.3, 45#, 1#, 1#, 12#, True)
			OpsPlnWinRatingB = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_B, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				28.3, 45#, 1#, 1#, 12#, True)
			OpsPlnWinRatingC = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_C, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				28.3, 45#, 1#, 1#, 12#, True)

			SprRatingA = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_A, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				31.7, 45#, 4#, 1#, 12#, True)
			SprRatingB = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_B, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				31.7, 45#, 4#, 1#, 12#, True)
			SprRatingC = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_C, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				31.7, 45#, 4#, 1#, 12#, True)

			SumRatingA = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_A, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				38.9, 45#, 7#, 1#, 12#, True)
			SumRatingB = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_B, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				38.9, 45#, 7#, 1#, 12#, True)
			SumRatingC = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_C, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				38.9, 45#, 7#, 1#, 12#, True)

			FallRatingA = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_A, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				35.6, 45#, 10#, 1#, 12#, True)
			FallRatingB = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_B, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				35.6, 45#, 10#, 1#, 12#, True)
			FallRatingC = TLineThermalModel.GetSteadyStateCurrentRating(152#, 32.234, tempC_C, diam_mm, 25#, 75#, rLow, rHigh, 0.91, 0.91, 0#, 0.61, _
				35.6, 45#, 10#, 1#, 12#, True)
		End If
	End If
	
	If condName <> "" Then ' Conductor exists
		Dim bundleFactor As String, spaceFactor As Double
		bundleFactor = 1
		spaceFactor = 1
		If Left(condName, 1) = "B" Or Left(condName, 2) = "TB" Then ' Bundled..must increase the rating based on spacing
			'Rounding on the conductor diameter is needed to prevent a derate due to 8" spacers being used in some parts of the system inside the substation. For example,
			'1272 ACSR has a diameter of 1.345". 6*1.345=8.07 and would cause a 10% derate. If this is rounded to the nearest whole number the result becomes 8 and no derate is required.
			'We don't want to derate it because of a 0.07" difference. Rob and Brian 8/18/2022
			If Not IsEmpty(BundledSpacing) Then
				If BundledSpacing < Round(6 * diam_in, 0) Then
					spaceFactor = 0.9
					If BundledSpacing < Round(3 * diam_in, 0) Then _
						spaceFactor = 0.75
				End If
			End If
			bundleFactor = 2
			If Left(condName, 2) = "TB" Then _
				bundleFactor = 3
		End If

		RatingA = RatingA * bundleFactor * spaceFactor
		RatingB = RatingB * bundleFactor * spaceFactor
		RatingC = RatingC * bundleFactor * spaceFactor
		WinRatingA = WinRatingA * bundleFactor * spaceFactor
		WinRatingB = WinRatingB * bundleFactor * spaceFactor
		WinRatingC = WinRatingC * bundleFactor * spaceFactor
		WinPeakRatingA = WinPeakRatingA * bundleFactor * spaceFactor
		WinPeakRatingB = WinPeakRatingB * bundleFactor * spaceFactor
		WinPeakRatingC = WinPeakRatingC * bundleFactor * spaceFactor
		OpsPlnWinRatingA = OpsPlnWinRatingA * bundleFactor * spaceFactor
		OpsPlnWinRatingB = OpsPlnWinRatingB * bundleFactor * spaceFactor
		OpsPlnWinRatingC = OpsPlnWinRatingC * bundleFactor * spaceFactor
		SprRatingA = SprRatingA * bundleFactor * spaceFactor
		SprRatingB = SprRatingB * bundleFactor * spaceFactor
		SprRatingC = SprRatingC * bundleFactor * spaceFactor
		SumRatingA = SumRatingA * bundleFactor * spaceFactor
		SumRatingB = SumRatingB * bundleFactor * spaceFactor
		SumRatingC = SumRatingC * bundleFactor * spaceFactor
		FallRatingA = FallRatingA * bundleFactor * spaceFactor
		FallRatingB = FallRatingB * bundleFactor * spaceFactor
		FallRatingC = FallRatingC * bundleFactor * spaceFactor

		Dim isTwoWinding As String
		isTwoWinding = ""
		If tempF = 500 Then
			isTwoWinding = "Amp_" ' Need to add this little string to ensure proper field is used - TwoWindingsSecitons is slightly different from BranchSections
		End If
		
		CalculateSectionRatings = isTwoWinding & "rate_A = " & Round(RatingA, 2) & ", " & _
			isTwoWinding & "rate_B = " & Round(RatingB, 2) & ", " & _
			isTwoWinding & "rate_C = " & Round(RatingC, 2) & ", " & _
			isTwoWinding & "rate_A_win = " & Round(WinRatingA, 2) & ", " & _
			isTwoWinding & "rate_B_win = " & Round(WinRatingB, 2) & ", " & _
			isTwoWinding & "rate_C_win = " & Round(WinRatingC, 2) & ", " & _
			isTwoWinding & "rate_A_WinPeak = " & Round(WinPeakRatingA, 2) & ", " & _
			isTwoWinding & "rate_B_WinPeak = " & Round(WinPeakRatingB, 2) & ", " & _
			isTwoWinding & "rate_C_WinPeak = " & Round(WinPeakRatingC, 2) & ", " & _
			isTwoWinding & "rate_A_OpsPlnWin = " & Round(OpsPlnWinRatingA, 2) & ", " & _
			isTwoWinding & "rate_B_OpsPlnWin = " & Round(OpsPlnWinRatingB, 2) & ", " & _
			isTwoWinding & "rate_C_OpsPlnWin = " & Round(OpsPlnWinRatingC, 2) & ", " & _
			isTwoWinding & "rate_A_Spr = " & Round(SprRatingA, 2) & ", " & _
			isTwoWinding & "rate_B_Spr = " & Round(SprRatingB, 2) & ", " & _
			isTwoWinding & "rate_C_Spr = " & Round(SprRatingC, 2) & ", " & _
			isTwoWinding & "rate_A_Sum = " & Round(SumRatingA, 2) & ", " & _
			isTwoWinding & "rate_B_Sum = " & Round(SumRatingB, 2) & ", " & _
			isTwoWinding & "rate_C_Sum = " & Round(SumRatingC, 2) & ", " & _
			isTwoWinding & "rate_A_Fall = " & Round(FallRatingA, 2) & ", " & _
			isTwoWinding & "rate_B_Fall = " & Round(FallRatingB, 2) & ", " & _
			isTwoWinding & "rate_C_Fall = " & Round(FallRatingC, 2)

    End If
End Function
