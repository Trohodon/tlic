Public Function WriteOutFullModel(FilePath As String, baseYear As Double, SnapshotIndex As Integer, useRegionalNumbering As Boolean, busOffset As Long, RegionalArea As Integer, _
    ZoneOffset As Integer, ZoneConfig As Integer, ownerOffset As Integer, overwriteSCEGArea As Boolean, SCEGAreaToOverwrite As Integer, AutoDispatch As Boolean, _
        SlackBus As Long, combineBuses As Boolean, ShortCircuit As Boolean, Dynamics As Boolean, UseOldWinterRatings As Boolean, TargetSwingGen As Double, _
            UseOpsPlnWinRatings As Boolean, UseSpringRatings As Boolean, DynamicDefaultRating As Boolean, GenDown As Boolean, SystemWinterPeakRatings As Boolean, _
                TRMImport As Boolean, TRMDec As Integer, TRMMDPV As Integer, TRMMDER As Integer, TRMIncSC As Integer, incSolar As Boolean, percentOfSolar As Integer, ignoreAfterQueueNum As Integer, _
                    incNoIA As Boolean, incAffSys As Boolean, incERIS As Boolean, incBusData As Boolean, incShuntData As Boolean, incBranchData As Boolean, incITCMData As Boolean, incLoadData As Boolean, _
                        incGenData As Boolean, writePython As Boolean, snapshot As Date, snapSeason As String, projectsSubset As Boolean, projectsStartDate As Date, _
                            projectsEndDate As Date, PSSEVer33Higher As Boolean) As Boolean

    WriteOutFullModel = False

    If useRegionalNumbering Then
        SlackBus = SlackBus + busOffset
    End If

    maxBus = 3000

    'Set global flags to passed in parameter:
    GenDownBool = GenDown

    Dim QDF As DAO.QueryDef
    Dim SQL As String

    Dim SnapshotName As String
    Dim SnapshotHeader1 As String
    Dim SnapshotHeader2 As String
    Dim SnapshotYear As Long
    Dim SnapshotSeason As String
    Dim SnapshotSystemLoad As Double
    Dim SnapshotInterchange As Double

    Dim VCS1Sch As Integer
    Dim VCS2Sch As Integer
    Dim VCS3Sch As Integer

    Dim SnapshotAngle As Double
    Dim File As Integer
    Dim SeqFile As Integer

    Dim Status As Boolean

    Dim OR_Interchange As Boolean
    OR_Interchange = False

    Dim BusSubsystem As DAO.Recordset
    Dim GenDownList As DAO.Recordset
    Dim TotalTRM As Double
    TotalTRM = 0

    Dim GD_TRMImport As Boolean
    GD_TRMImport = False

    Dim TextFile As Integer, AffSysFilePath As String, FileContent As String 'For Affected System file loading

    'Get Ref to DB:
    Set db = CurrentDb()

    'Get Subsystem Recordset and cache into array:
    Set BusSubsystem = db.OpenRecordset("BusesToExport", dbOpenDynaset, dbSeeChanges)

    SubsystemSize = DCount("bus", "BusesToExport")
    ReDim BusSubsys(1 To SubsysSize) As Long

    BusSubsystem.MoveLast
    BusSubsystem.MoveFirst
    For i = 1 To SubsysSize
        BusSubsys(i) = BusSubsystem.Fields("bus").Value
    BusSubsystem.MoveNext
    Next i

    'Get GenDownList Recordset and cache into array:
    Set GenDownList = db.OpenRecordset("GenToShutDown", dbOpenDynaset, dbSeeChanges)
    GenDownSize = DCount("Gen", "GenToShutDown")
    On Error Resume Next
    ReDim GenDownArr(1 To GenDownSize) As Long
    ReDim GenDownTRMArr(1 To GenDownSize) As Boolean

    GenDownList.MoveLast
    GenDownList.MoveFirst
    i = 1
    For i = 1 To GenDownSize
        GenDownArr(i) = GenDownList.Fields("Gen").Value
    GenDownTRMArr(i) = GenDownList.Fields("ReplGDwTRM").Value
    GenDownList.MoveNext
    Next i

    Dim projectsBuses As DAO.Recordset, projectsBranches As DAO.Recordset, projectsMachines As DAO.Recordset, projectsSwitchedShunts As DAO.Recordset
    Dim projectsTwoWindings As DAO.Recordset

    Set projectsBuses = db.OpenRecordset("ProjectsBuses", dbOpenDynaset, dbSeeChanges)
    Set projectsBranches = db.OpenRecordset("ProjectsBranches", dbOpenDynaset, dbSeeChanges)
    Set projectsMachines = db.OpenRecordset("ProjectsMachines", dbOpenDynaset, dbSeeChanges)
    Set projectsSwitchedShunts = db.OpenRecordset("ProjectsSwitchedShunts", dbOpenDynaset, dbSeeChanges)
    Set projectsTwoWindings = db.OpenRecordset("ProjectsTwoWindings", dbOpenDynaset, dbSeeChanges)

    ReDim projectsBusesArr(0)
    ReDim projectsBranchesArr(0)
    ReDim projectsMachinesArr(0)
    ReDim projectsSwitchedShuntsArr(0)
    ReDim projectsTwoWindingsArr(0)
    ReDim projectsBusesArrFinDate(0)
    ReDim projectsBranchesArrFinDate(0)
    ReDim projectsMachinesArrFinDate(0)
    ReDim projectsSwitchedShuntsArrFinDate(0)
    ReDim projectsTwoWindingsArrFinDate(0)

    If projectsSubset Then
        SnapshotYear = Year(snapDate)
    SnapShotSeason = snapSeason
    snapshotDate = snapDate

    Call ProjectArrayCreate(projectsStartDate, projectsEndDate, projectsBuses)
    Call ProjectArrayCreate(projectsStartDate, projectsEndDate, projectsBranches)
    Call ProjectArrayCreate(projectsStartDate, projectsEndDate, projectsMachines)
    Call ProjectArrayCreate(projectsStartDate, projectsEndDate, projectsSwitchedShunts)
    Call ProjectArrayCreate(projectsStartDate, projectsEndDate, projectsTwoWindings)
    Else
        On Error GoTo -1
        'get snapshot info:
        Set Snapshots = db.OpenRecordset("Snapshots", dbOpenDynaset, dbSeeChanges)
        Snapshots.MoveLast
        Snapshots.MoveFirst
        While Not Snapshots.EOF
            If Snapshots.Fields("model_index") = SnapshotIndex Then
                SnapshotName = Snapshots.Fields("model_name")
                SnapshotYear = Snapshots.Fields("year")

                SnapShotSeason = Snapshots.Fields("season")
                snapshotDate = Snapshots.Fields("date")
                SnapshotSystemLoad = Snapshots.Fields("system_load_MW")
                'Added for CF Study- loads CF variables from Snapshots and loads SnapshotDate Buffer
                CFStudy = Snapshots.Fields("Contingent Facility")
                CFStudyDate = Snapshots.Fields("ConFacDate")
                SnapshotDateBuffer = snapshotDate

                If Not IsNull(Snapshots.Fields("OR_interchange_MW")) Then
                    OR_Interchange = True
                    SnapshotInterchange = Snapshots.Fields("OR_interchange_MW")
                End If
                SnapshotAngle = Snapshots.Fields("default_angle_deg")
                SnapshotHeader1 = Snapshots.Fields("psse_header1")
                SnapshotHeader2 = Snapshots.Fields("psse_header2")
            End If
            Snapshots.MoveNext
        Wend
        Snapshots.Close

        'get Schedules (Interchange) info:
        If Not OR_Interchange Then
            SnapshotInterchange = 0
            Set Schedules = db.OpenRecordset("Schedules(Interchange)", dbOpenDynaset, dbSeeChanges)
            Schedules.MoveLast
            Schedules.MoveFirst
            While Not Schedules.EOF
                If Schedules.Fields("Start") < snapshotDate And snapshotDate < Schedules.Fields("End") Then
                    If SnapShotSeason = "L" Then
                        If Not Schedules.Fields("Peak_only") Then
                            SnapshotInterchange = SnapshotInterchange + Schedules.Fields("MW")
                            If Left(Schedules.Fields("Sch"), 4) = "VCS1" Then
                                VCS1Sch = Schedules.Fields("MW")
                            ElseIf Left(Schedules.Fields("Sch"), 4) = "VCS2" Then
                                VCS2Sch = Schedules.Fields("MW")
                            ElseIf Left(Schedules.Fields("Sch"), 4) = "VCS3" Then
                                VCS3Sch = Schedules.Fields("MW")
                            End If
                        End If
                    Else
                        SnapshotInterchange = SnapshotInterchange + Schedules.Fields("MW")
                        If Left(Schedules.Fields("Sch"), 4) = "VCS1" Then
                            VCS1Sch = Schedules.Fields("MW")
                        ElseIf Left(Schedules.Fields("Sch"), 4) = "VCS2" Then
                            VCS2Sch = Schedules.Fields("MW")
                        ElseIf Left(Schedules.Fields("Sch"), 4) = "VCS3" Then
                            VCS3Sch = Schedules.Fields("MW")
                        End If
                    End If
                End If
        End If

    Schedules.MoveNext
    Wend
    Schedules.Close
    End If
    End If 'If...Else projectsSubset Then

    Dim BaseSeason As String
    If SnapShotSeason = "W" Or Left(SnapShotSeason, 1) = "D" Or SnapShotSeason = "H" Or SnapShotSeason = "L" Then
        BaseSeason = Left(SnapShotSeason, 1)
        'percentOfSolar = 0 'Shouldn't be necessary to do here anymore, since it's done in the ModelBuildForm code
    Else
        BaseSeason = "S"
    End If

    'create python file:
    File = FreeFile
    If writePython Then
        'create python file:
        Open FilePath & ".raw.py" For Output As #File
        commChar = "#"
    Else
        'create aux file:
        Open FilePath & ".aux" For Output As #File
        commChar = "//"
    End If

    If ShortCircuit Then
        'create Seq file:
        SeqFile = FreeFile
        Open FilePath & ".seq.py" For Output As #SeqFile

        Print #SeqFile, commChar & " PSSE Sequence (Short Circuit) Model Data for SCEG Control Area" & vbCrLf & commChar & " Auto-Generated Snapshot from SCEGModelData DB"
        Print #SeqFile, commChar & " File Created: " & Now() & vbCrLf & commChar & " Snapshot Year: " & SnapshotYear
        Print #SeqFile, commChar & " Snapshot Season: " & SnapShotSeason & vbCrLf & commChar & " Snapshot Date: " & snapshotDate
    End If

    'Print file metadata:
    Print #File, commChar & " PSSE Powerflow Model Data for SCEG Control Area" & vbCrLf & commChar & " Auto-Generated Snapshot from SCEGModelData DB"
    Print #File, commChar & " File Created: " & Now() & vbCrLf & commChar & " Snapshot Year: " & SnapshotYear
    Print #File, commChar & " Snapshot Season: " & SnapShotSeason & vbCrLf & commChar & " Snapshot Date: " & snapshotDate
    If projectsSubset Then
        Print #File, commChar & " Projects subset dates - Start: " & projectsStartDate & " End: " & projectsEndDate
    Else
        Print #File, commChar & " Snapshot RetailLoad: " & SnapshotSystemLoad & " MW" & vbCrLf & commChar & " Snapshot Net Interchange: " & SnapshotInterchange & " MW"
    End If

    'Added for CF Study- Adds CF Study to header
    If CFStudy Then
        Print #File, commChar & " This File is a Contingent Facilities Study"
    End If

    ' If we are overwriting an existing SCEG area, describe to user and delete old area:

    If OverwriteSCEGArea And incBusData Then
        'Added Or ShortCircuit to print out the following when short circuit model is printed but loads are not included AU 6 5 18
    'If OverwriteSCEGArea And incBusData And incShuntData And incBranchData And incTfmrData And incLoadData And incGenData Then
        Print #File, ""
        If writePython Then
            If Not projectsSubset And incShuntData And incBranchData And incTfmrData And (incLoadData Or ShortCircuit) And incGenData Then
                Print #File, commChar & " This file includes a script that removes a preexisting SCEG area from the world case"
                Print #File, "sceg_area_to_overwrite = " & SCEGAreaToOverwrite & vbCrLf & "psspy.bsys(1,0,[1,0,765.],1,[sceg_area_to_overwrite],0,[],0,[],0,[])"
            Else
                Print #File, "sceg_area = " & SCEGAreaToOverwrite & vbCrLf & "psspy.bsys(1,0,[1,0,765.],1,[sceg_area],0,[],0,[],0,[])"
            End If

            ' Get average bus angle from current case
            ' ierr, rarray = abusreal(sid, flag, string)
            '     Integer SID    Defines the bus subsystem to be used (input: -1 by default).
            '                    SID = a negative value, to instruct the API to assume a subsystem containing all buses in the working case.
            '                    SID = a valid bus subsystem identifier. Valid subsystem identifiers range from 0 to 11. Subsystem SID must have been previously defined.
            '     Integer FLAG   Is a flag indicating which subsystem buses to include (input: 1 by default).
            '                    FLAG = 1, for only in-service buses.
            '                    FLAG = 2, for all buses.
            '     Character      Is an array of NSTR element specifying NSTR of the following strings indicating the bus quantities desired (input: no default allowed):
            ' STRING(NSTR)
            '
            '                    'BASE'   Bus base voltage, in kV.
            '                    'PU'     Actual bus voltage magnitude, in pu.
            '                    'KV'     Actual bus voltage magnitude, in kV (in pu if base voltage is 0.0).
            '                    'ANGLE'  Bus voltage phase angle, in radians.
            '                    'ANGLED' Bus voltage phase angle, in degrees.
            '                    'NVLMHI' Normal high voltage limit, in pu.
            '                    'NVLMLO' Normal low voltage limit, in pu.
            '                    'EVLMHI' Emergency high voltage limit, in pu.
            '                    'EVLMLO' Emergency low voltage limit, in pu.
            '                    'MISMATCH' Bus mismatch in MVA (0.0 if bus type code > 3).
            '                    'O_MISMATCH' Bus mismatch in pu determined by the power output option setting (0.0 if bus type code > 3).

            Print #File, vbCrLf & "' Attain average bus angle " & vbCrLf
            Print #File, "busAngles = BusAngles[0] # the above api dumps the id data into a 2d array, so extract the first element with an in-place replacement"
            Print #File, "i = 0 " & vbCrLf & "avgBusAngle = 0 " & vbCrLf & "while i < len(busAngles):" & vbCrLf & vbTab & "avgBusAngle += busAngles[i]"
            Print #File, vbTab & "i += 1 " & vbCrLf & "avgBusAngle /= len(busAngles)" ' Added to make a list of buses in case 6/25/19 AU

            Print #File, "currentBus=0 " & vbCrLf & "angleDictionary={}" & vbCrLf & "voltageDictionary={}" & vbCrLf & "newBuses=[]"

            '!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!Mod Starts Here

            Set Buses = db.OpenRecordset("Buses", dbOpenDynaset, dbSeeChanges)
            Buses.MoveLast
            Buses.MoveFirst

            Dim BusList As String

    BusList = "busList = ["

    While Not Buses.EOF
        'If InStr(Buses.Fields("reduced_name"), "_STAR") = 0 Then 'Added for 3 W mod to prevent star bus writing in python script
    If Buses.Fields("Include") And Buses.Fields("bus_number") <> 0 And BusInSubsystem(Buses.Fields("bus_number")) Then ' Populates Array of bus numbers 6/25/19 AU, UPDATE 8/13/19

        If Int(Buses.Fields("bus_number")) = Buses.Fields("bus_number") Then
            If Buses.Fields("area_number") < 344 And Buses.Fields("bus_number") < 10000 And Buses.Fields("base_kV") <> 1 Then 'Added "Buses.Fields("base_kV") <> 1 to exclude 3W star buses AU 2/11/25
                BusList = BusList & Buses.Fields("bus_number") + 3700000 & ","
            Else
                BusList = BusList & Buses.Fields("bus_number") & ","
            End If
        End If

    '!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
    End If
    'Else
    'MsgBox Buses.Fields("reduced_name")
    'End If 'If InStr(B... 'Added for 3 W mod to prevent star bus writing in python script

    Buses.MoveNext
    Wend

    BusList = BusList & "]"
    Print #File, BusList

    Print #File, "def busCheck(busNum):" & vbCrLf & vbTab & "global angleDictionary" & vbCrLf & vbTab & "global avgBusAngle" & vbCrLf & vbTab & "global newBuses"
    Print #File, vbTab & "global voltageDictionary" & vbCrLf & vbTab & "ierr, angle = psspy.busdat(busNum,'ANGLE')" & vbCrLf & vbTab & "ierr, voltage = psspy.busdat(busNum,'PU')"
    Print #File, vbTab & "ierr, angle = psspy.busdat(busNum,'ANGLED')" & vbCrLf & vbTab & "angleDictionary[busNum]=angle"
    Print #File, vbTab & "ierr, voltage = psspy.busdat(busNum,'PU')" & vbCrLf & vbTab & "voltageDictionary[busNum]=voltage"
    Print #File, vbTab & "else:" & vbCrLf & vbTab & vbTab & "angleDictionary[busNum]=avgBusAngle" & vbCrLf & vbTab & vbTab & "voltageDictionary[busNum]=1.0"
    Print #File, vbTab & "newBuses.append(busNum)" & vbCrLf & vbTab & "def retBus(busNum):"
    Print #File, vbTab & "global currentBus" & vbCrLf & vbTab & "currentBus=busNum" & vbCrLf & vbTab & "return busNum" & vbCrLf & vbCrLf & "for bus in busList:"
    Print #File, vbTab & "busCheck(bus)" & vbCrLf

    '''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''
    'Mod Ends Here 6/25/19 AU

    If Not projectsSubset And incShuntData And incBranchData And incTfmrData And (incLoadData Or ShortCircuit) And incGenData Then
        Print #File, "psspy.extr(1,0,[0,0])" & vbCrLf & "psspy.purgarea(0, [1])"
    End If
    Else
        Print #File, commChar & " This file includes a script that removes a preexisting SCEG area from the world case:" & vbCrLf & "SCRIPT" & vbCrLf & "{"
    Print #File, "EnterMode(Edit); // Powerworld must be in edit mode to edit devices" & vbCrLf & "Delete(Bus, ""SCEG*"");" & vbCrLf & "}"
    End If 'If writePython Then
    End If 'If OverwriteSCEGArea And incBusData Then

    'write headers:
    If writePython And Not projectsSubset Then
        Print #File, "psspy.case_title_data('" & SnapshotHeader1 & "','" & SnapshotHeader2 & "')"
    ElseIf Not projectsSubset Then
        Print #File, vbCrLf & "PWCaseInformation (Selected)" & vbCrLf & "{" & vbCrLf & "  " & vbCrLf & "  <SUBDATA PWCaseHeader> " & vbCrLf & "  //Case Description"
    Print #File, "  SnapshotHeader1 " & vbCrLf & "  " & SnapshotHeader2 & vbCrLf & "  </SUBDATA>" & vbCrLf & "}"
    End If

    Print #File, "DATA (Area, [AreaNum,AreaName,BGAGC,BusSlack]) {" & vbCrLf & "(" & IIf(OverwriteSCEGArea, SCEGAreaToOverwrite & ",""SCEG"" ,""Area Slack"" ," & SlackBus & "", "")", vbCrLf & ");"

    Print #File, "Owner (Number,Name) {" & vbCrLf & "(" & vbCrLf & " 361 ""SCEG""" & vbCrLf & ");"
    End If

    If Not projectsSubset Then
        ' Have to do Load data work before doing Generator data before printing them both out. We need to determine bus types (1 for regular bus, 2 for gen bus)
    Dim loadLine As String

    'Turn off warnings so access doesn't complain to user about creating/overwriting a new table:
    DoCmd.SetWarnings False

    'create new table called 'Loads' via SQL from 'LoadPointsWithBuses' query.
    SQL = "SELECT LoadPointsWithBuses.load_point_id, LoadPointsWithBuses.Scalable, LoadPointsWithBuses.pti_bus, LoadPointsWithBuses.pti_load, LoadPointsWithBuses.description, " & _
          "LoadPointsWithBuses.include, LoadPointsWithBuses.trans_r, LoadPointsWithBuses.trans_x, LoadPointsWithBuses.Wholesale, LoadPointsWithBuses.NativeLoad, " & _
              "LoadPointsWithBuses.k_seasonal, LoadPointsWithBuses.zone_loc, LoadPointsWithBuses.zone_city, LoadPointsWithBuses.zone_load, " & _
                  "LoadPointsWithBuses.parent_bus_number, LoadPointsWithBuses.start_date, LoadPointsWithBuses.end_date, " & _
                      "LoadPointsWithBuses.comment INTO Loads FROM LoadPointsWithBuses;"
    DoCmd.RunSQL SQL

    'add column called 'MW'
    SQL = "ALTER TABLE Loads ADD COLUMN MW DOUBLE;"
    DoCmd.RunSQL SQL

    'add column called 'MVAR'
    SQL = "ALTER TABLE Loads ADD COLUMN MVAR DOUBLE;"
    DoCmd.RunSQL SQL

    'Be responsible and turn warnings back on:
    DoCmd.SetWarnings True

    'Open newly created load table and history/forecast tables also:
    Set Loads = db.OpenRecordset("Loads", dbOpenDynaset, dbSeeChanges)
    Set LoadHistory = db.OpenRecordset("LoadHistory", dbOpenDynaset, dbSeeChanges)
    Set LoadForecast = db.OpenRecordset("LoadForecast", dbOpenDynaset, dbSeeChanges)

    'Reset Cursors:
    Loads.MoveLast
    Loads.MoveFirst
    LoadHistory.MoveLast
    LoadHistory.MoveFirst
    LoadForecast.MoveLast
    LoadForecast.MoveFirst

    Dim Break As Boolean, loadHisIsNull As Boolean
    Dim MW As Double        'Final MW value after overall 'case' scaling
    Dim MWO As Double       'Initial MW value before overall 'case' scaling
    Dim MVAR As Double
    Dim QoverP As Double
    Dim Pwholesale As Double
    Dim P_non_native As Double    'non native load is load that is not included in CorpForecast. Ex. Orangeburg Load points. 2024-02-06 BU
    Dim Pretail As Double
    Dim ID As Long
    Dim bus As Long
    Dim YearNum As Integer
    Dim YearCnt As Integer
    Dim RetailK, TotalLoad, VCS1SS As Double

    'Loop through load points and populate with initial loads:
    DoCmd.SetWarnings False
    Pwholesale = 0#
    P_non_native = 0#
    Pretail = 0#
    While Not Loads.EOF

        If (Loads.Fields("include").Value And Loads.Fields("pti_bus") < maxBus) Or (Dynamics And Loads.Fields("pti_bus") < 740 And _
            Loads.Fields("pti_bus") > 722) Then   'VCS1 SS buses between 722 and 740
    If Loads.Fields("start_date") <= snapshotDate And Loads.Fields("end_date") > snapshotDate Then
        'If Left(Loads.Fields("pti_load").Value, 1) <> "P" Or (Left(Loads.Fields("pti_load").Value, 1) = "P" And Not IsNull(Loads.Fields("Queue").Value) And _
            'Loads.Fields("Queue").Value <= ignoreAfterQueueNum And (Loads.Fields("SignedIA") Or incNoIA) And incERIS) Then  '2018-12-31 JWR - Got rid of this - all solar is model with machines, not negative loads, now.
        ID = Loads.Fields("load_point_id").Value

        'get base load:
        Break = False
        loadHisIsNull = False
        MW = 0#
        MVAR = 0#

        Dim backYear As Integer
        backYear = 0
        Do While Not Break And backYear < 4   ' Go back 3 years
            LoadHistory.MoveFirst
            Do While Not LoadHistory.EOF And Not Break
                If LoadHistory.Fields("load_year") = baseYear - backYear And LoadHistory.Fields("load_point_id") = ID Then
                    If LoadHistory.Fields("load_season") = BaseSeason Then
                        If IsNull(LoadHistory.Fields("MW")) Then
                            loadHisIsNull = True
                            Exit Do  ' Hop out of loop - the MW value is null, meaning we need previous year's
                        End If

                        MW = LoadHistory.Fields("MW").Value
                        MVAR = LoadHistory.Fields("MVAR").Value
                        QoverP = 0
                        If MW = 0 Then
                            'you can't divide by zero. If MW=0 then assign a default PF of .98
                            QoverP = 0.20305    'estimate PF = .98
                        Else
                            'otherwise get a Q/P so the correct MVAR value can be calculated later
                            QoverP = MVAR / MW
                        End If
                        Break = True
                    ElseIf BaseSeason <> "S" And LoadHistory.Fields("load_season") = "S" Then
                        'If IsNull(LoadHistory.Fields("MW")) Then
                        '   loadHisIsNull = True
                        '   Exit Do ' Hop out of loop - the MW value is null, meaning we need previous year's
                        'End If
                    End If
                End If

                LoadHistory.MoveNext
            Loop

            backYear = backYear + 1
        Loop

        backYear = 0
        Do While Not Break And backYear < 6   ' Go back 5 years
            If Not loadHisIsNull And SnapShotSeason <> "S" And Not Break Then ' Non-summer value doesn't exist, so must loop back through and get the summer value
                LoadHistory.MoveFirst
                While Not LoadHistory.EOF And Not Break
                    If LoadHistory.Fields("load_year") = baseYear - backYear And LoadHistory.Fields("load_point_id") = ID And LoadHistory.Fields("load_season") = "S" Then
                        MW = LoadHistory.Fields("MW").Value

    MVAR = LoadHistory.Fields("MVAR").Value
    QoverP = 0
    If MW = 0 Then
        'you can't divide by zero. If MW=0 then assign a default PF of 0.98
    QoverP = 0.20305    'estimate PF = .98
    Else
        'otherwise get a Q/P so the correct MVAR value can be calculated later
    QoverP = MVAR / MW
    End If
    Break = True

    Dim loadScale As Double
    loadScale = 1

    If Loads.Fields("Scalable").Value Then
        If BaseSeason = "D" Then
            loadScale = 0.43
    ElseIf BaseSeason = "L" Then
        loadScale = 0.35
    ElseIf BaseSeason = "H" Then
        loadScale = 0.8
    End If
    End If

    MW = MW * loadScale
    If loadScale <> 1 And MVAR < 0 Then _
        MVAR = MVAR * loadScale   ' I don't think it's a good idea to scale absolute value of MVAR down, thereby increasing MVAR, if MVAR is negative for lighter load periods - JWR
    End If

    Dim forecastMW As Double, forecastFrac
    forecastMW = LoadForecast.Fields("dMW").Value
    forecastFrac = LoadForecast.Fields("dFrac").Value
    'Scale the forecast data by a seasonal multiplier
    If Loads.Fields("Scalable").Value Then
        If BaseSeason = "D" Then
