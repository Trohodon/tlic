                    backYear = 0

                                    MVAR = LoadHistory.Fields("MVAR").Value

                                        QoverP = MVAR / MW
                                    End If

                                    Break = True

                                End If
                            End If

                            LoadHistory.MoveNext
                        Loop

                        backYear = backYear + 1

                                End If
                                LoadHistory.MoveNext
                            Wend

        Print #File, commChar & "===================== TRANSFORMER DATA ========================"

        Print #File, "Branch (BusNum,BusNum:1,LineCircuit, LineR,LineX, LineAMVA,LineAMVA:1,LineAMVA:2,LineXFType,XFAuto,XFRegBus,LineXfmr,LineStatus,BranchDeviceType,XFMVABase,XFRegMax,XFRegMin,XFConfiguration,XFFixedTap,XFFixedTap:1,XFNominalKV:1,XFTapMin,XFTapMax)" & vbCrLf & "{"

            While Not Transformers.EOF

                        Exit For
                    End If

                    Bus2 = Transformers.Fields("to_bus")

Dim XFAuto As String
