Public Sub CalculateAllImpedancesAndRatings()
    Dim db As DAO.Database 'Database handle
    Dim Sections As DAO.Recordset 'Recordset object for Section Table
    Dim NotFound As String ' List of sections not found
    Dim SomeNotFound As Boolean 'True if at least one section not found
    Dim Success As Boolean 'Return value from CalculateSectionImpedances
    Success = False
    Dim SectionID As Long 'Section id field cache
    Dim SectionName As String 'Section name field cache
    Dim Result As VbMsgBoxResult 'Confirm execute msg box result

    Set db = CurrentDb 'Get handle to current database
    Set Sections = db.OpenRecordset("BranchSections") 'Open BranchSections Query
    NotFound = "The following section IDs were not found:" & vbCrLf 'Init error string
    SomeNotFound = False 'Init error flag

    'Reset row cursor
    Sections.MoveLast
    Sections.MoveFirst

    Result = MsgBox("This will take a few minutes", vbOKCancel, "ImpCalc Routine") 'Warn user of time requirements
    If Result = vbOK Then 'If user is OK with time then:
        DoCmd.SetWarnings False 'Turn off warnings
        If Not (Sections.EOF And Sections.EOF) Then 'If table is not empty then:
            'Loop through sections, calculate section impedance, and write impedances back to section record:
            While Not Sections.EOF 'Until the end of the table:
                SectionID = Sections.Fields("section_id").Value 'Get the section id from the current row
                SectionName = Sections.Fields("section_name").Value 'Get the section name from the current row
                If IsNumeric(Left(SectionName, 1)) Or UCase(SectionName) = "NONAME" Or UCase(Left(SectionName, 2)) = "CD" _
                    Or UCase(Left(SectionName, 3)) = "BUS" Or UCase(Left(SectionName, 4)) = "FOLD" Then ' Only calc ratings for conductors (their sect name starts with number, NONAME, CD (Ckt breaker Drop)), BUS or FOLD (Fold-in)
                    'If IsNumeric(Left(SectionName, 1)) Or SectionName = "NONAME" Or Left(SectionName, 2) = "CD" Then ' Only calc for conductors (their sect name starts with number or CD (Ckt breaker Drop))
                    RatingsCalc.tempF = Sections.Fields("MOT_°F").Value
                    RatingsCalc.phase_cond_ID = Sections.Fields("phase_cond_id").Value
                    RatingsCalc.RebuiltSoon = Sections.Fields("RebuiltSoon").Value
                    If IsNumeric(Sections.Fields("BundledSpacing").Value) Then ' If num value present assign it
                        RatingsCalc.BundledSpacing = Sections.Fields("BundledSpacing").Value
                    Else ' Otherwise set it higher than 12
                        BundledSpacing = 13
                    End If

                    Dim SQL As String, calcRatingsString As String
                    SQL = "UPDATE BranchSections SET "
                    calcRatingsString = CalculateSectionRatings(False)
                    If calcRatingsString <> "" Then
                        'Build SQL string for ratings
                        SQL = SQL & calcRatingsString
                    End If

                    Dim calcImpedancesString As String
                    If IsNumeric(Left(SectionName, 1)) Or SectionName = "NONAME" Then ' Only calc impedances for conductors (their sect name starts with number)
                        calcImpedancesString = CalculateSectionImpedances(SectionID, True) 'Calculate impedances
                        If calcImpedancesString <> "" Then
                            If calcRatingsString <> "" Then _
                                SQL = SQL & ", "
                            'Build SQL string for impedances
                            SQL = SQL & calcImpedancesString
                        ElseIf calcImpedancesString = vbCancel Then
                            DoCmd.SetWarnings True 'turn warnings back on
                            Exit Sub
                        End If
                    End If

                    SQL = SQL & " WHERE section_id = " & SectionID
                    If calcRatingsString <> "" And calcImpedancesString <> "" Then
                        DoCmd.RunSQL SQL
                    End If
                End If

                'If calcImpedancesString = "" Then 'If unsuccessful, add line to error report:
                'NotFound = NotFound & SectionID & vbCrLf
                'SomeNotFound = True
                'End If
                Sections.MoveNext 'Move cursor to next row
            Wend
        End If
		
		    'Now calculate Transformer impedances if the 'Estimate' or 'Calculate' option is chosen
        Dim XFMRs As DAO.Recordset 'Recordset object for TwoWindings Table
        Dim XFMRLosses As DAO.Recordset 'Recordset object for EstimateTransformerLosses Table
        Dim XFMRBusesQuery As DAO.Recordset 'Recordset object for TwoWindingWithBusses Query
        Dim TwoWindingID As Integer

        ' need to use variant or empty records will produce a type mismatch error
        Dim CalcType As Variant
        Dim Z0_CalcType As Variant
        Dim PowerBase As Variant ' variable for the mva_base column
        Dim XFMRImped As Variant 'variable for the Z% column
        Dim XFMRRateA As Variant
        Dim FL_Cop_Loss As Variant
        Dim XFMR_R As Variant
        Dim XFMR_X As Variant
        Dim XFMR_NomVolt As Variant
        Dim ZpuOld As Variant 'only needed as a placeholder for impedance calculation
        Dim Bus1NomVolt As Variant
        Dim Bus2NomVolt As Variant

        'Dim BusID As Variant
        'Dim EstOA_Rating As Double
        'Dim EstXpu As Double 'estimating Zpu or really the X since R is so small in an xfmr.
        'Dim EstRpu As Double
        'Dim EstLosses As Variant 'estimating losses so a calculation of R and X can happen

        DoCmd.SetWarnings False

        Set XFMRs = db.OpenRecordset("TwoWindings") 'Open TwoWindings Table
        Set XFMRBusesQuery = db.OpenRecordset("TwoWindingsWithBusses") 'Open TwoWindingWithBusses Query
        Set XFMRLosses = db.OpenRecordset("EstimateTransformerLosses", dbOpenDynaset, dbSeeChanges) 'Open EstimateTransformerLosses Query
        XFMRBusesQuery.MoveFirst

        While Not XFMRBusesQuery.EOF 'Do until the end of the table:
            TwoWindingID = XFMRBusesQuery.Fields("two_winding_id").Value 'Get the section id from the current row
            CalcType = XFMRBusesQuery.Fields("R_X_CalcType").Value
            Z0_CalcType = XFMRBusesQuery.Fields("ZeroSeq_CalcType").Value
            PowerBase = XFMRBusesQuery.Fields("mva_base").Value 'assuming the mva_base is the OA rating of the xfmr
            XFMRImped = XFMRBusesQuery.Fields("Z%").Value
            FL_Cop_Loss = XFMRBusesQuery.Fields("FL_Copper_Losses").Value
            XFMRRateA = XFMRBusesQuery.Fields("RateA").Value
            XFMR_NomVolt = XFMRBusesQuery.Fields("WindingTwoNomVolt").Value
            Bus1NomVolt = XFMRBusesQuery.Fields("Bus1_base_kV").Value
            Bus2NomVolt = XFMRBusesQuery.Fields("Bus2_base_kV").Value

            If CalcType = "Estimate" And Not IsNull(XFMRRateA) Then 'I at least need the size/rateA to estimate XFMR R and X.
                XFMR_R = Round(DLookup("[Rpu]", "EstimateTransformerLosses", "Size_MVA = " & XFMRBusesQuery.Fields("RateA")), 6)
                XFMR_X = Round(DLookup("[Xpu]", "EstimateTransformerLosses", "Size_MVA = " & XFMRBusesQuery.Fields("RateA")), 6)

                If IsNull(XFMR_R) Or IsNull(XFMR_X) Then  'If the size could not be found in the EstimateTransformerLosses table use a default impedance
                'Not using and Estimate calculation right now due to low R values causing squart root of a negative number. Using fixed default in EstimateTransformerLosses table.
                'EstOA_Rating = XFMRRateA * 0.7 'general rule of thumb for estimating the OA Rating
                'EstXpu = Round(7.8 / EstOA_Rating, 10) 'estimating Zpu or really the X since R is so small in an xfmr.
                'EstRpu = Sqr((7.8 ^ 2) / (EstOA_Rating ^ 2) - (EstXpu ^ 2)) 'back calculate R given an estimation of X
                XFMR_R = Round(DLookup("[Rpu]", "EstimateTransformerLosses", "Size_MVA = 22.4 "), 6) 'most common distribution transformer
                XFMR_X = Round(DLookup("[Xpu]", "EstimateTransformerLosses", "Size_MVA = 22.4 "), 6) 'most common distribution transformer
                DoCmd.RunSQL "UPDATE TwoWindings SET R = " & XFMR_R & ", X = " & XFMR_X & " WHERE (two_winding_id = " & TwoWindingID & ");" 'place new values in table
				
                Else 'a size was found in the EstimateTransformerLosses table so use what it found
                    DoCmd.RunSQL "UPDATE TwoWindings SET R = " & XFMR_R & ", X = " & XFMR_X & " WHERE (two_winding_id = " & TwoWindingID & ");" 'place new values in table
                End If

            ElseIf CalcType = "Calculated" And Not IsNull(PowerBase) And Not IsNull(FL_Cop_Loss) And Not IsNull(XFMRImped) And Not IsNull(XFMR_NomVolt) Then 'need all four to calculate R and X from test report
                If Bus2NomVolt > Bus1NomVolt Then
                    Bus1NomVolt = Bus2NomVolt 'We need the System High Side bus voltage per the formula so this makes sure we get it.
                End If

                XFMR_R = Round((FL_Cop_Loss / 10000) / (PowerBase ^ 2), 6)
                ZpuOld = Round((XFMRImped / 100) / PowerBase * 100, 6) 'this assumes the system base is always 100 MVA and is hard coded as shown
                XFMR_X = Round(ZpuOld * (XFMR_NomVolt ^ 2 / Bus1NomVolt ^ 2), 7) 'using the change of base formula here.

                DoCmd.RunSQL "UPDATE TwoWindings SET R = " & XFMR_R & ", X = " & XFMR_X & " WHERE (two_winding_id = " & TwoWindingID & ");" 'place new values in table

            ElseIf CalcType = "Override" Then 'get override value so it can be used for zero sequence as well
                XFMR_R = XFMRBusesQuery.Fields("R").Value
                XFMR_X = XFMRBusesQuery.Fields("X").Value

            Else 'if the CalcType field is left blank or proper cells were not filled in then make the field blank
            
			DoCmd.RunSQL "UPDATE TwoWindings SET R = Null, X = Null WHERE (two_winding_id = " & TwoWindingID & ");" 'set R and X to blank if the R_X_CalcType field is left blank

            XFMR_R = XFMRBusesQuery.Fields("R").Value 'set variable to null so zero seq can be set to blank as well
            XFMR_X = XFMRBusesQuery.Fields("X").Value 'set variable to null so zero seq can be set to blank as well

            If XFMRBusesQuery.Fields("include") = True Then 'if the field is blank and the include box is checked...send user a warning message
                MsgBox "MUST CHECK! Transformer with two_winding_id " & TwoWindingID & " has no impedance. Make sure proper data is provided for the desired CalcType", vbOKOnly + vbCritical, "Warning" 'warn the user
            End If
        End If

        'now address zero sequence columns
        If Z0_CalcType = "UsePosSeq" Then 'check to see if it's blank or an error will occur
            If IsNull(XFMR_R) Or IsNull(XFMR_X) Then 'if its blank(meaning the user forgot to put override values in, then make the zero seq field blank
                DoCmd.RunSQL "UPDATE TwoWindings SET R0_pu = Null, X0_pu = Null WHERE (two_winding_id = " & TwoWindingID & ");" 'set R0 and X0 to blank if the override values are left blank
            Else
                DoCmd.RunSQL "UPDATE TwoWindings SET R0_pu = " & XFMR_R & ", X0_pu = " & XFMR_X & " WHERE (two_winding_id = " & TwoWindingID & ");" 'place new values in table
            End If

        'no action needed if Z0_CalcType = Override. The code in the ModelWriter will use the values there.
        Else 'no action needed
        End If

        XFMRBusesQuery.MoveNext 'Move cursor to next row
    Wend

    '***********************************************************************************************************
    'Print errors to user:
    If SomeNotFound Then 'If one or more sections did not compute then:
        MsgBox "These sections did not have their impedances calculated:" & vbCrLf & NotFound 'print errors
    Else
        MsgBox "Done with no errors."
    End If

    DoCmd.SetWarnings True 'turn warnings back on
	
	'now update the 'TimeStamp' table so the time stamp text box on the ModelBuildForm is updated below each button.
    DoCmd.SetWarnings False
    DoCmd.RunSQL "UPDATE TimeStamp1 SET TimeStamp1.Recalc_Impedances_And_Ratings = " & Chr(39) & Format(Now(), "MM/dd/yyyy  hh:mm AM/PM") & Chr(39) & " WHERE TimeStamp_ID = " & 1 & ";"
    DoCmd.RunSQL "UPDATE TimeStamp1 SET TimeStamp1.Recalc_Impedances = " & Chr(39) & Format(Now(), "MM/dd/yyyy  hh:mm AM/PM") & Chr(39) & " WHERE TimeStamp_ID = " & 1 & ";"
    DoCmd.RunSQL "UPDATE TimeStamp1 SET TimeStamp1.Recalc_Ratings = " & Chr(39) & Format(Now(), "MM/dd/yyyy  hh:mm AM/PM") & Chr(39) & " WHERE TimeStamp_ID = " & 1 & ";"
    DoCmd.SetWarnings True

End If

End Sub

Public Sub CalculateAllImpedances()
    Dim db As DAO.Database 'Database handle
    Dim Sections As DAO.Recordset 'Recordset object for Section Table
    Dim NotFound As String ' List of sections not found
    Dim SomeNotFound As Boolean 'True if at least one section not found
    Dim Success As Boolean 'Return value from CalculateSectionImpedances
    Success = False
    Dim SectionID As Long 'Section id field cache
    Dim SectionName As String 'Section name field cache
    Dim Result As VbMsgBoxResult 'Confirm execute msg box result

    Set db = CurrentDb 'Get handle to current database
    Set Sections = db.OpenRecordset("BranchSections") 'Open BranchSections Query
    NotFound = "The following section IDs were not found:" & vbCrLf 'Init error string
    SomeNotFound = False 'Init error flag

    'Reset row cursor
    Sections.MoveLast
    Sections.MoveFirst

    If Not (Sections.EOF And Sections.EOF) Then 'If table is not empty then:
        Result = MsgBox("This will take a few minutes", vbOKCancel, "ImpCalc Routine") 'Warn user of time requirements
        If Result = vbOK Then 'If user is OK with time then:
            DoCmd.SetWarnings False 'Turn off warnings
            'Loop through sections, calculate section impedance, and write impedances back to section record:
            While Not Sections.EOF 'Until the end of the table:
                SectionID = Sections.Fields("section_id").Value 'Get the section id from the current row
                SectionName = Sections.Fields("section_name").Value 'Get the section name from the current row

                Dim SQL As String, calcImpedancesString As String
                SQL = "UPDATE BranchSections SET "
                If IsNumeric(Left(SectionName, 1)) Or SectionName = "NONAME" Then ' Only calc impedances for conductors (their sect name starts with number)
                    calcImpedancesString = CalculateSectionImpedances(SectionID, True) 'Calculate impedances
                    If calcImpedancesString <> "" And Not IsNumeric(calcImpedancesString) Then
                        'Build SQL string for impedances
                        SQL = SQL & calcImpedancesString & " WHERE section_id = " & SectionID
                    ElseIf IsNumeric(calcImpedancesString) Then
                        If calcImpedancesString = vbCancel Then
                            DoCmd.SetWarnings True 'turn warnings back on
                            Exit Sub
                        End If
                    End If
                Else
                    calcImpedancesString = ""
                End If

                If calcImpedancesString <> "" Then _
                    DoCmd.RunSQL SQL 'Run SQL

                ' If Not Success Then 'If unsuccessful, add line to error report:
                '     NotFound = NotFound & SectionID & vbCrLf
                '     SomeNotFound = True
                ' End If
                Sections.MoveNext 'Move cursor to next row
            Wend
            'Print errors to user:
            If SomeNotFound Then 'If one or more sections did not compute then:
                MsgBox "These sections did not have their impedances calculated:" & vbCrLf & NotFound 'print errors
            Else
                MsgBox "Done with no errors."
            End If
            DoCmd.SetWarnings True 'turn warnings back on
        End If
    End If
    'now update the 'TimeStamp' table so the time stamp text box on the ModelBuildForm is updated below each button.
    DoCmd.SetWarnings False
    DoCmd.RunSQL "UPDATE TimeStamp1 SET TimeStamp1.Recalc_Impedances = " & Chr(39) & Format(Now(), "MM/dd/yyyy  hh:mm AM/PM") & Chr(39) & " WHERE TimeStamp_ID = " & 1 & ";"
    DoCmd.SetWarnings True
End Sub

Public Function CalculateSectionImpedances(SectionID As Long, Verbose As Boolean) As String
    CalculateSectionImpedances = ""

    Dim ans As String

    'Db Objects:
    Dim db As DAO.Database 'Database Handle
    Dim QDF As DAO.QueryDef 'Query Definition Object
    Dim Sections As DAO.Recordset 'Recordset for Section Data

    'Impedance Values:
    Dim Impedances As Variant 'Return Variable for Line.CalculateImpedance Function
    Dim R1, X1, B1, R0, X0, B0 As Double
    Dim Params(21) As Double 'Input data array for Line.CalculateImpedance Function

    ' Dim line As tlinelib.TLine 'Init Line object. This object is contained in a COM-wrapped .NET assembly
    'This call will not work unless the tlielib.dll is registered on your machine using regasm.exe

    'Impedance Calc Params:
    Dim Ax, Ay, Bx, By, Cx, Cy, G1x, G1y, G2x, G2y As Double 'Structure Position Coords
    Dim PhaseGMR, PhaseRadius, PhaseR, PhaseX, PhaseC As Double 'Conductor properties
    Dim StaticGMR, StaticRadius, StaticR, StaticX, StaticC As Double 'Static wire properties
    Dim KV, Mileage, Rho As Double 'corridor parameters
    Dim Zfactor, Yfactor As Double 'these will cache factor to convert SI impedances to p.u.

    'Open section data init Line:
    Set db = CurrentDb 'Set the Db variable to point to the current Db
    Set QDF = db.QueryDefs("SectionsWithData") 'Run canned query that combines all relevant section data
    Set Sections = QDF.OpenRecordset 'open the query data as a recordset object
    ' Set ElectricalLineModel = New tlinelib.TLine 'instantiate Line object
    Set ElectricalLineModel = New TLICs.TLine_JWR 'instantiate Line object

    'Dim Rating As Variant
    'Set ThermalLineModel = New TLineCalc.TLineThermalModel
    'Rating = ThermalLineModel.GetSteadyStateCurrentRating(150#, 32.25, 100#, 15#, 25#, 75#, 0.001, 0.0015, 0.91, 0.91, 0#, 6.1, 35#, 45#, 1#, 7#, 1#, 12#)

    ' CalculateSectionImpedances = True

    ' Find record:
    Sections.MoveLast 'Reset cursor
    Sections.MoveFirst
    Sections.FindFirst ("section_id = " & SectionID) 'Find the row associated with the passed in SectionID

    On Error Resume Next
    If Not Sections.NoMatch Then 'Record Found:
        'get Data from Section Table:
        Ax = Sections.Fields("ax").Value
        Ay = Sections.Fields("ay").Value
        Bx = Sections.Fields("bx").Value
        By = Sections.Fields("by").Value
        Cx = Sections.Fields("cx").Value
        Cy = Sections.Fields("cy").Value
        G1x = Sections.Fields("g1x").Value
        G1y = Sections.Fields("g1y").Value
        G2x = Sections.Fields("g2x").Value
        G2y = Sections.Fields("g2y").Value
        PhaseR = Sections.Fields("phase_r").Value
        PhaseX = Sections.Fields("phase_x").Value
        PhaseC = Sections.Fields("phase_c").Value
        StaticR = Sections.Fields("static_r").Value
        StaticX = Sections.Fields("static_x").Value
        StaticC = Sections.Fields("static_c").Value
        PhaseGMR = Sections.Fields("phase_gmr").Value
        StaticGMR = Sections.Fields("static_gmr").Value
        PhaseRadius = Sections.Fields("phase_radius").Value
        StaticRadius = Sections.Fields("static_radius").Value
        KV = Sections.Fields("base_kV").Value
        Mileage = Sections.Fields("mileage").Value
        Rho = Sections.Fields("rho").Value

        If Mileage > 0 And StaticGMR <> 0 And StaticGMR <> "" And Not IsNull(StaticGMR) Then
            'Call Line.CalculateImpedances Method of object TLine in library tlinelib
            'Inputs: [Ax(ft), Ay(ft), Bx(ft), By(ft), Cx(ft), Cy(ft), G1x(ft), G1y(ft), G2x(ft), G2y(ft), PhaseGMR(ft), PhaseRadius(ft), PhaseR(ohm/mi), PhaseX(ohm/mi/ft), PhaseC(Mohm-mi/ft), StaticGMR(ft), StaticRadius(ft), StaticR(ohm/mi), Rho(ohm-m)]
            'Outputs: [R1(ohms/mi), X1(ohms/mi), B1(uS/mi), R0(ohms/mi), X0(ohms/mi), B0(uS/mi)]:

            Params(0) = Ax
            Params(1) = Ay
            Params(2) = Bx
            Params(3) = By
            Params(4) = Cx
            Params(5) = Cy
            Params(6) = G1x
            Params(7) = G1y
            Params(8) = G2x
            Params(9) = G2y
            Params(10) = PhaseGMR
            Params(11) = PhaseRadius
            Params(12) = PhaseR
            Params(13) = PhaseX
            Params(14) = PhaseC
            Params(15) = StaticGMR
            Params(16) = StaticRadius
            Params(17) = StaticR
            Params(18) = Mileage
            Params(19) = Rho
            Params(20) = KV
            Params(21) = 100 ' 100 MVA base conversion

            ' Impedances = ElectricalLineModel.CalculateImpedances(Ax, Ay, Bx, By, Cx, Cy, G1x, G1y, G2x, G2y, _
            ' PhaseGMR, PhaseRadius, PhaseR, PhaseX, PhaseC, StaticGMR, StaticRadius, StaticR, Rho)

            Impedances = ElectricalLineModel.CalculateImpedances(Params)

            'Calculate inverses of per unit impedance and admittance bases
            'Zfactor = 1/Zbase, Yfactor = 1E-6/Ybase (admittance returned is in [uS/mi])
            ' Zfactor = Mileage * 100 / (KV * KV)   '(mi*Sbase/Vbase^2) i.e. ohm/mi to pu on 100 MVA base conversion
            ' Yfactor = Mileage * 0.00000001 * KV * KV '(mi*1E-6*Vbase^2/Sbase) i.e. Mohm-mi to pu on 100 MVA base conversion
        End If
		
		        If Not IsEmpty(Impedances) Then
            If UBound(Impedances) = 11 Then 'Check dimensions of returned array
                'Extract values from return object and convert to per unit:

                R1 = Impedances(0) ' * Zfactor
                If Impedances(1) < 0 Or Impedances(1) > 1 Or Impedances(4) < 0 Or Impedances(4) > 3 Or StaticGMR = 0 Then
                    X1 = 0
                    B1 = 0
                    R0 = 0
                    X0 = 0
                    B0 = 0
                Else
                    X1 = Impedances(1) ' * Zfactor
                    B1 = Impedances(2) ' * Yfactor
                    R0 = Impedances(3) ' * Zfactor
                    X0 = Impedances(4) ' * Zfactor
                    B0 = Impedances(5) ' * Yfactor
                End If

                '      'Build SQL string to write impedances to section table:
                '      SQL = "UPDATE BranchSections SET " & _
                '      "r1_pu = " & Round(R1, 6) & ", " & _
                '      "x1_pu = " & Round(X1, 6) & ", " & _
                '      "b1_pu = " & Round(B1, 6) & ", " & _
                '      "r0_pu = " & Round(R0, 6) & ", " & _
                '      "x0_pu = " & Round(X0, 6) & ", " & _
                '      "b0_pu = " & Round(B0, 6) & _
                '      " WHERE section_id = " & SectionID
                '
                '      DoCmd.RunSQL SQL 'Run SQL:

                CalculateSectionImpedances = "r1_pu = " & Round(R1, 6) & ", x1_pu = " & Round(X1, 6) & ", b1_pu = " & Round(B1, 6) & ", r0_pu = " & Round(R0, 6) & ", x0_pu = " & _
                    Round(X0, 6) & ", b0_pu = " & Round(B0, 6)
            Else 'Wrong dimensions in return impedance array
                If Verbose And StaticGMR <> 0 And StaticGMR <> "" And Not IsNull(StaticGMR) Then
                    MsgBox "ImpCalc Error: TLine.CalculateImpedances Function failed for Section: " & SectionID
                End If
            End If
        ElseIf Mileage = 0 Then
            CalculateSectionImpedances = "r1_pu = 0, x1_pu = 0, b1_pu = 0, r0_pu = 0, x0_pu = 0, b0_pu = 0" ' Just set all to zero"
        Else 'return impedance array is empty
            If Verbose Then
                ans = MsgBox("ImpCalc Error: TLine.CalculateImpedances Function failed for Section: " & SectionID, vbOKCancel)
                If ans = vbCancel Then
                    CalculateSectionImpedances = ans
                End If
            End If
        End If

    Else 'No Match, Bad SectionID!
        If Verbose Then
            ans = MsgBox("ImpCalc Error: provided Section ID (" & SectionID & ") does not exist.", vbOKCancel)
            If ans = vbCancel Then
                CalculateSectionImpedances = ans
            End If
        End If
    End If

    On Error GoTo -1
End Function
