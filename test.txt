    End If

    '''''''Added for CF Study- Adds CF Study to header
    If CFStudy Then
        Print #File, commChar & " This File is a Contingent Facilities Study"
    End If

    'If we are overwriting an existing SCEG area, describe to user and delete old area:

    If OverwriteSCEGArea And incBusData Then
        'Added Or ShortCircuit to print out the following when short circuit model is printed but loads are not included AU 6 5 18

        Print #File, ""
        'Create Filters
		
        Print #File, "Data(Filter, [ObjectType, FilterName, FilterLogic, Number, FilterPre, Enabled, DataMaintainerAssign], AUXDEF, YES)" & vbCrLf & "{"
        'Print #File, "" & Chr(34) & "Bus" & Chr(34) & " " & Chr(34) & " " & Chr(34) & "AND" & Chr(34) & " " & Chr(34) & "1" & Chr(34) & " " & Chr(34) & "NO" & Chr(34) & " " & Chr(34) & "YES" & Chr(34)
        'Print #File, "" & Chr(34) & "Branch" & Chr(34) & " " & Chr(34) & " " & Chr(34) & "AND" & Chr(34) & " " & Chr(34) & "1" & Chr(34) & " " & Chr(34) & "NO" & Chr(34) & " " & Chr(34) & "YES" & Chr(34)
        'Print #File, "" & Chr(34) & "Gen" & Chr(34) & " " & Chr(34) & " " & Chr(34) & "AND" & Chr(34) & " " & Chr(34) & "1" & Chr(34) & " " & Chr(34) & "NO" & Chr(34) & " " & Chr(34) & "YES" & Chr(34)
        Print #File, Chr(34) & "Load" & Chr(34) & " " & Chr(34) & "SCEG" & Chr(34) & " " & Chr(34) & " AND " & Chr(34) & " " & "1" & " " & Chr(34) & "NO" & " " & Chr(34) & " " & Chr(34) & "YES" & Chr(34) & " " & Chr(34) & Chr(34) & vbCrLf & vbTab & Chr(60) & "SUBDATA Condition" & Chr(62) & vbCrLf & vbTab & " AreaName startswith " & Chr(34) & "SCEG" & Chr(34) & vbCrLf & vbTab & Chr(60) & Chr(47) & "SUBDATA" & Chr(62)

        Print #File, "" & Chr(34) & "Gen" & Chr(34) & "," & Chr(34) & "SCEG" & Chr(34) & "," & Chr(34) & " " & Chr(34) & "AND" & Chr(34) & "," & Chr(34) & "1" & Chr(34) & "," & Chr(34) & "NO" & Chr(34) & "," & Chr(34) & "YES" & Chr(34)

        Print #File, Chr(34) & "Gen" & Chr(34) & "," & Chr(34) & "GEN ON" & Chr(34) & "," & Chr(34) & "Closed" & Chr(34) & vbCrLf & vbTab & "GenID notstartswith " & Chr(34) & "R" & Chr(34) & vbCrLf & vbTab & Chr(60)

        & "1" & " " & Chr(34) & "NO" & " " & Chr(34) & " " & Chr(34) & "YES" & Chr(34) & " " & Chr(34) & Chr(34) & vbCrLf _
        & vbTab & Chr(60) & "SUBDATA Condition" & Chr(62) & vbCrLf & vbTab & " AreaName startswith " & Chr(34) & "SCEG" & Chr(34) & vbCrLf _

            & vbTab & " BusName notcontains " & Chr(34) & "1FFP HY7" & Chr(34) & vbCrLf _

        Print #File, "}"

        'Save existing Bus and Gen data
        Print #File, commChar & " This file includes a script that removes a preexisting SCEG area from the world case:" & vbCrLf & "SCRIPT" & vbCrLf & "{"
        'Removed 2_14_20 AU

        'Turn off warnings so access doesn't complain to user about creating/overwriting a new table:
        DoCmd.SetWarnings False

        'create new table called 'Loads' via SQL from LoadPointsWithBuses query.
        SQL = "SELECT LoadPointsWithBuses.load_point_id, LoadPointsWithBuses.pti_bus, LoadPointsWithBuses.pti_load, LoadPointsWithBuses.description, " & _
            "LoadPointsWithBuses.include, LoadPointsWithBuses.Scalable, LoadPointsWithBuses.Interruptible, LoadPointsWithBuses.wholesale, LoadPointsWithBuses.NativeLoad, " & _

        Set Loads = db.OpenRecordset("Loads", dbOpenDynaset, dbSeeChanges)
        Set LoadHistory = db.OpenRecordset("LoadHistory", dbOpenDynaset, dbSeeChanges)
        Set LoadForecast = db.OpenRecordset("LoadForecast", dbOpenDynaset, dbSeeChanges)

        'Reset Cursors:
        Loads.MoveLast
        Loads.MoveFirst
        LoadHistory.MoveLast
        LoadHistory.MoveFirst
        LoadForecast.MoveLast
        LoadForecast.MoveFirst

        Dim Break As Boolean, loadHisIsNull As Boolean
        Dim MW As Double              'Final MW value after overall 'case' scaling
        Dim MW0 As Double             'Initial MW value before overall 'case' scaling
        Dim MVAR As Double
        Dim QoverP As Double
        Dim Pwholesale As Double
        Dim P_non_native As Double    'non native load is load that is not included in CorpForecast. Ex. Orangeburg Load points. 2024-02-06 BU
        Dim Pretail As Double
        Dim ID As Long
        Dim bus As Long
        Dim YearNum As Integer
        Dim YearCnt As Integer
        Dim RetailK, TotalLoad, VCS1SS As Double

        'Loop through load points and populate with initial loads:
        DoCmd.SetWarnings False
        Pwholesale = 0#
        P_non_native = 0#
        Pretail = 0#

        While Not Loads.EOF

            If (Loads.Fields("include").Value And Loads.Fields("pti_bus") < maxBus) Or (Dynamics And Loads.Fields("pti_bus") < 740 And _
                Loads.Fields("pti_bus") > 722) Then 'VCS1SS buses between 722 and 740

                If Loads.Fields("start_date") <= snapshotDate And Loads.Fields("end_date") > snapshotDate Then

                    ' If Left(Loads.Fields("pti_load").Value, 1) <> "P" Or (Left(Loads.Fields("pti_load").Value, 1) = "P" And incSolar And Not IsNull(Loads.Fields("Queue").Value) And _
                        ' Loads.Fields("Queue").Value <= ignoreAfterQueueNum And (Loads.Fields("signedIA") Or incNoIA) And incERIS) Then '2018-12-31 JWR - Got rid of this - all solar is model with machines, not negative loads, now.

                        ID = Loads.Fields("load_point_id").Value

                        'get base load:
                        Break = False
                        loadHisIsNull = False
                        MW = 0#
                        MVAR = 0#

                        Dim backYear As Integer
                        backYear = 0

                        Do While Not Break And backYear < 4 'Go back 3 years

                            LoadHistory.MoveFirst

                            Do While Not LoadHistory.EOF And Not Break

                                If LoadHistory.Fields("load_year") = baseYear - backYear And LoadHistory.Fields("load_point_id") = ID Then

                                    If LoadHistory.Fields("load_season") = BaseSeason Then

                                        If IsNull(LoadHistory.Fields("MW")) Then
                                            loadHisIsNull = True
                                            Exit Do 'Hop out of loop - the MW value is null, meaning we need previous years
                                        End If

                                        MW = LoadHistory.Fields("MW").Value
                                        MVAR = LoadHistory.Fields("MVAR").Value

                                        QoverP = 0

                                        If MW = 0 Then    'you can't divide by zero. If MW=0 then assign a default PF of 0.98
                                            QoverP = 0.20305    'estimate PF = .98
                                        Else 'otherwise get a Q/P so the correct MVAR value can be calculated later
                                            QoverP = MVAR / MW
                                        End If

                                        Break = True

                                    End If
                                End If

                                LoadHistory.MoveNext
                            Loop

                            backYear = backYear + 1
                        Loop

                        backYear = 0
                        Do While Not Break And backYear < 6 'Go back 5 years
                            If Not loadHisIsNull And SnapShotSeason <> "S" And Not Break Then 'Non-summer value doesn't exist, so must loop back through and get the summer value
                                LoadHistory.MoveFirst
                                While Not LoadHistory.EOF And Not Break
                                    If LoadHistory.Fields("load_year") = baseYear - backYear And LoadHistory.Fields("load_point_id") = ID And LoadHistory.Fields("load_season") = "S" Then
                                        MW = LoadHistory.Fields("MW").Value
                                        MVAR = LoadHistory.Fields("MVAR").Value
                                        QoverP = 0
                                        If MW = 0 Then    'you can't divide by zero. If MW=0 then assign a default PF of 0.98
                                            Exit Do 'Hop out of loop - the MW value is null, meaning we need previous years
                                        Else    'otherwise get a Q/P so the correct MVAR value can be calculated later
                                            QoverP = MVAR / MW
                                        End If
                                        Break = True
										
                                        Dim loadScale As Double
                                        loadScale = 1

                                        If Loads.Fields("Scalable").Value Then
                                            If BaseSeason = "D" Then
                                                loadScale = 0.43
                                            ElseIf BaseSeason = "L" Then
                                                loadScale = 0.35
                                            ElseIf BaseSeason = "H" Then
                                                loadScale = 0.8
                                            End If
                                        End If

                                        MW = MW * loadScale

                                        If loadScale <> 1 And MVAR < 0 Then _
                                            MVAR = MVAR * loadScale  ' I don't think it's a good idea to scale absolute value of MVAR down, thereby increasing MVAR, if MVAR is negative for lighter load periods - JWR
										
                                    End If
                                    LoadHistory.MoveNext
                                Wend
                            End If

                            backYear = backYear + 1
                        Loop

                        'check to see if MVAR is much higher than MW for each load point.
                        'QoverP = 0#     'this is in the wrong place. Moved it up  BKU 2019-09-09
                        If MW > 0# Or MW < 0# Then
                            'QoverP = MVAR / MW 'this is in the wrong place. Moved it up BKU 2019-09-09
                            If QoverP > 10 Then
                                'MsgBox "The MVAR / MW ratio is very high (" & QoverP & ") for load point ID " & ID & ". Setting ratio to 10 to prevent issues." 'Removed for MUltiple Models to prevent Pop-up. May reinstate as logged errror AU 4/13/2020
                                QoverP = 10
                            End If
                        End If

                        If month(GenMaintenance.Fields("start_date")) <= 7 And month(GenMaintenance.Fields("end_date")) >= 7 Then
                            Dispatchable = False
                            Pgen = GenMaintenance.Fields("output_MW").Value

                            Stat = 0
                        End If
                    ElseIf SnapShotSeason = "W" Then
                        If month(GenMaintenance.Fields("start_date")) <= 1 And month(GenMaintenance.Fields("end_date")) >= 1 Then
                            Dispatchable = False
                            Pgen = GenMaintenance.Fields("output_MW").Value
                            Stat = 0
                        End If
                    ElseIf SnapShotSeason = "F" Then
                        If month(GenMaintenance.Fields("start_date")) <= 10 And month(GenMaintenance.Fields("end_date")) >= 10 Then
                            Dispatchable = False
                            Pgen = GenMaintenance.Fields("output_MW").Value
                            Stat = 0
                        End If
                    ElseIf SnapShotSeason = "Z" Or SnapShotSeason = "L" Or SnapShotSeason = "LL" Then

                            Dispatchable = False
                            Pgen = GenMaintenance.Fields("output_MW").Value
                            Stat = 0

                    End If
                End If

            'if not active element (future or past)
            If snapshotDate < StartDate Or snapshotDate > EndDate Then
                Available = False
            Else

                If GenDownListPos(GenBus) <= GenDownSize Then
                    Stat = 0

                    If GenDownTRMArr(GenDownListPos(GenBus)) Then
                        GD_TRMImport = True

                        If GenBus = 370803 Then ' VCS1 offline, so interchange must be adjusted
                            SnapshotInterchange = SnapshotInterchange - Pgen + VCS1Sch
                            TotalTRM = TotalTRM + Pgen - VCS1Sch
                        ElseIf GenBus = 370835 Then 'VCS2 offline, so interchange must be adjusted
                            SnapshotInterchange = SnapshotInterchange - Pgen + VCS2Sch
                            TotalTRM = TotalTRM + Pgen - VCS2Sch
                        ElseIf GenBus = 370836 Then 'VCS3 offline, so interchange must be adjusted
                            SnapshotInterchange = SnapshotInterchange - Pgen + VCS3Sch
                            TotalTRM = TotalTRM + Pgen - VCS3Sch
                        Else

                            TotalTRM = TotalTRM + Pgen
                        End If
                    Else

                            SnapshotInterchange = SnapshotInterchange - VCS1Sch
                        ElseIf GenBus = 370835 Then 'VCS2 offline, so interchange must be adjusted	
                            SnapshotInterchange = SnapshotInterchange - VCS2Sch
                        ElseIf GenBus = 370836 Then 'VCS3 offline, so interchange must be adjusted
                            SnapshotInterchange = SnapshotInterchange - VCS3Sch

                If Available And Dispatchable And AutoDispatch And Stat <> 0 Then

                        Pgen = TotalGen - CurrentGen

                        If GenMet Or (Left(Generators.Fields("pti_machine_id"), 1) = "E" And Not incERIS) Then
                            Stat = 0
                        Else
                            If SnapShotSeason = "L" And GenBus < 370829 And GenBus > 370820 Then
                                Pgen = -85
                            ElseIf Pgen <= 0 Then
                                Stat = 0
                            End If

                            CurrentGen = CurrentGen + Pgen
                            If CurrentGen + TargetSwingGen > TotalGen Then
                                Pgen = TotalGen - CurrentGen + Pgen - TargetSwingGen
                                CurrentGen = TotalGen - TargetSwingGen
                                GenMet = True
                        End If
                    End If
                End If
            End If

            'Machine:
            If Not Available Then
                Stat = Available
            End If

            'Added PWDGenLine for PW gen aux file Mod AU 12 2
            If Stat = 0 Then
                StatString = "Open"
            Else
                StatString = "Closed"
            End If

            PWDgenLine = PWDgenLine & GenBus & " " & RemBus & " " & Chr(34) & Trim(genID) & Chr(34) & " " & Chr(34) & StatString & Chr(34) & " " & Format(Pgen, "0.000") & " " & "0.000" & " " _

            If genArray(0) <> 0 Then _
                ReDim Preserve genArray(UBound(genArray) + 1)
                genArray(UBound(genArray)) = GenBus

            'Seq Data:
            If ShortCircuit Then
                genSeqLine = genSeqLine & "psspy.seq_machine_data(" & GenBus & ", '" & Trim(genID) & "', ["

                'Print #SeqFile, genSeqLine
            End If
        End If
    End If
